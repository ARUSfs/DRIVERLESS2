157void Controller::on_speed_timer()
158{
159 if(!(pointsXY_.empty()) && run_check_){
160 get_global_index();
161
162 if (braking_procedure_){
163 target_speed_ = std::max(0.0, std::sqrt(vx_*vx_ - 2*kBrakingDecc*target_speed_));
164 } else if(!(speed_profile_.empty())){
165 target_speed_ = speed_profile_.at(index_global_);
166 } else{
167 target_speed_ = kTargetSpeed;
168 }
169 std_msgs::msg::Float32 target_speed_msg;
170 target_speed_msg.data = target_speed_;
171 target_speed_pub_->publish(target_speed_msg);
172
173 double target_acc = 0.0;
174 if(!(acc_profile_.empty())){
175 target_acc = acc_profile_.at(index_global_);
176 }
177
178
179 double dt = (this->now() - previous_time_).seconds();
180 acc_cmd_ = speed_control_.get_acc_command(target_speed_, target_acc, vx_, dt);
181 previous_time_ = this->now();
182
183 common_msgs::msg::Cmd cmd;
184 cmd.acc = std::clamp(acc_cmd_, kMinCmd, kMaxCmd);
185 if (vx_ < 3) cmd.acc = std::clamp(acc_cmd_, 0.0, kMaxCmd);
186 cmd.delta = std::clamp(delta_cmd_, -kMaxSteer*M_PI/180, kMaxSteer*M_PI/180);
187 cmd.target_r = lti_mpc_.target_r_;
188 cmd_pub_ -> publish(cmd);
189 }
190}

193void Controller::on_steer_timer()
194{
195 if ((!optimized_ && kFirstLapSteerControl=="PP") || (optimized_ && kOptimizedSteerControl=="PP")){
196 pure_pursuit_.set_path(pointsXY_);
197 pure_pursuit_.set_position(x_, y_, yaw_);
198
199 if (!optimized_) {
200 pure_pursuit_.get_steering_angle(index_global_, kLAD);
201 } else {
202 pure_pursuit_.get_steering_angle(index_global_, kOptLAD);
203 }
204 delta_cmd_ = pure_pursuit_.delta_cmd_;
205 Point pursuit_point = pure_pursuit_.pursuit_point_;
206
207 common_msgs::msg::PointXY pursuit_point_msg;
208 pursuit_point_msg.x = pursuit_point.x;
209 pursuit_point_msg.y = pursuit_point.y;
210 pursuit_point_pub_ -> publish(pursuit_point_msg);
211
212 } else if ((!optimized_ && kFirstLapSteerControl=="MPC") || (optimized_ && kOptimizedSteerControl=="MPC"))
213 {
214 Point position;
215 position.x = x_;
216 position.y = y_;
217 if (!(s_.empty())){
218 double t0 = this->now().seconds();
219
220 if(!(speed_profile_.empty())){
221 lti_mpc_.set_reference_trajectory(pointsXY_, speed_profile_, s_, position, yaw_, vx_, index_global_);
222 delta_cmd_ = lti_mpc_.calculate_control(delta_, v_delta_, vy_, r_);
223 }
224
225 if (kDebug) RCLCPP_INFO(this->get_logger(), "MPC time: %f ms", (this->now().seconds() - t0)*1000);
226 }
227 }

259void Controller::trajectory_callback(const common_msgs::msg::Trajectory::SharedPtr msg)
260{
261 if (optimized_ || msg->points.size() < 3){
262 return;
263 }
264
265 // Check if the trajectory is the same
266 if(msg->points.size() != pointsXY_.size()
267 || msg->points[msg->points.size()-1].x != pointsXY_[pointsXY_.size()-1].x
268 || msg->points[msg->points.size()-1].y != pointsXY_[pointsXY_.size()-1].y ){
269 new_trajectory_ = true;
270
271 pointsXY_.clear();
272
273 std::vector<common_msgs::msg::PointXY> points_common = msg -> points;
274 for (const auto &pointXY : points_common) {
275 Point point;
276 point.x = pointXY.x;
277 point.y = pointXY.y;
278 pointsXY_.push_back(point);
279 }
280 s_= msg -> s;
281 k_ = msg -> k;
282 speed_profile_ = msg -> speed_profile;
283 acc_profile_ = msg -> acc_profile;
284 }

300void Controller::get_global_index() {
301 double min_dist = std::numeric_limits<double>::max();
302 int N = pointsXY_.size();
303
304 // If the trajectory is not updated, the search starts from the last index
305 int i;
306 if(new_trajectory_){
307 i = 0;
308 } else {
309 i = index_global_;
310 }
311
312 // Search for the closest point from the current position
313 // N*1.1 and i%N is added to allow restarting loops (trackdrive)
314 while (i < int(N*1.1)) {
315 double dx = pointsXY_[i % N].x - x_;
316 double dy = pointsXY_[i % N].y - y_;
317 double dist = dx * dx + dy * dy;
318
319 if (dist < min_dist) {
320 min_dist = dist;
321 index_global_ = i % N;
322 } else if(!new_trajectory_ && dist > 5.0){
323 // If the distance is greater than 5 meters from last index,
324 // the search is stopped to avoid errors due to loops in the trajectory (skidpad)
325 break;
326 }
327 i++;
328 }
329 new_trajectory_ = false;
330}
