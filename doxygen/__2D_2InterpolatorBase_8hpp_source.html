136 template <typename XIter, typename YIter, typename ZIter>
137 auto setData(const XIter& x_begin, const XIter& x_end, const YIter& y_begin,
138 const YIter& y_end, const ZIter& z_begin, const ZIter& z_end)
139 -> decltype(std::copy(x_begin, x_end, xData.begin()),
140 std::copy(y_begin, y_end, yData.begin()),
141 std::copy(z_begin, z_end, zData.begin()), void()) {
142 xData.clear();
143 yData.clear();
144 zData.clear();
145
146 xData.reserve(z_end - z_begin);
147 yData.reserve(z_end - z_begin);
148 zData.reserve(z_end - z_begin);
149
150 if ((x_end - x_begin) == (z_end - z_begin) &&
151 (y_end - y_begin) == (z_end - z_begin)) {
152 std::copy(x_begin, x_end, std::back_inserter(xData));
153 std::copy(y_begin, y_end, std::back_inserter(yData));
154 std::copy(z_begin, z_end, std::back_inserter(zData));
155 } else if ((x_end - x_begin) * (y_end - y_begin) == (z_end - z_begin)) {
156 std::copy(z_begin, z_end, std::back_inserter(zData));
157 for (auto it = x_begin; it != x_end; it++) {
158 std::fill_n(std::back_inserter(xData), y_end - y_begin, *it);
159 std::copy(y_begin, y_end, std::back_inserter(yData));
160 }
161
162 } else {
163 throw std::runtime_error(
164 "Interpolator data format is not supported. The x, y, and z "
165 "data containers should all be the same length, or the length "
166 "of the z container should be equal to the product of the "
167 "length of the x and y containers.");
168 }
169
170 this->setUnsafeDataReference(xData.size(), xData.data(), yData.data(),
171 zData.data());
172 }

327template <class Derived, typename Real>
328void InterpolatorBase<Derived, Real>::setup2DDataViews() {
329 // make sure data has actually been initialized
330 if (!xView || !yView || !zView) return;
331
332 // setup 2D view of the data
333 // We need to figure out what the x and y dimensions are.
334 int N = zView->size();
335 int Nx = 0, Ny = 0;
336 // Ny will be the number of elements that have the same x coordinate
337 Real xlast = (*xView)(0);
338 while (Ny < N - 1 && fabs((*xView)(Ny)-xlast) < 1e-40) Ny++;
339 Nx = N / Ny;
340
341 // consecutive values in the x data are separated by Ny, so this is the
342 // inner stride for X
343 X.reset(new _2DVectorView(xView->data(), Nx,
344 Eigen::InnerStride<Eigen::Dynamic>(Ny)));
345
346 // consecutive values in the y data are next to each other, so the stride is
347 // just 1
348 Y.reset(new _2DVectorView(yView->data(), Ny,
349 Eigen::InnerStride<Eigen::Dynamic>(1)));
350
351 // Eigen defaults to COLUMN MAJOR
352 // consecutive elements in a column are separated by Ny (this is the inner
353 // stride) consecutive elements in a row are located next to each other
354 // (this is the outer stride) Stride object takes outer,inner as arguments.
355 Z.reset(new _2DMatrixView(
356 zView->data(), Nx, Ny,
357 Eigen::Stride<Eigen::Dynamic, Eigen::Dynamic>(1, Ny)));
358}
