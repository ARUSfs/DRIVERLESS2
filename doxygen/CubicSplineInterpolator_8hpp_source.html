72template <class Real>
73Real CubicSplineInterpolator<Real>::operator()(Real x) const {
74 BASE::checkData();
75
76 const MapType& X = *(this->xView);
77 const MapType& Y = *(this->yView);
78
79 // don't extrapolate at all
80 if (x < X(0) || x > X(X.size() - 1)) return 0;
81
82 int i = this->get_index_to_right_of(x);
83
84 // See the wikipedia page on "Spline interpolation"
85 // (https://en.wikipedia.org/wiki/Spline_interpolation) for a derivation
86 // this interpolation.
87 Real t = (x - X(i - 1)) / (X(i) - X(i - 1));
88 Real q = (1 - t) * Y(i - 1) + t * Y(i) +
89 t * (1 - t) * (a[i - 1] * (1 - t) + b[i - 1] * t);
90
91 return q;
92}

94template <typename Real>
95Real CubicSplineInterpolator<Real>::derivative(Real x) const {
96 const MapType& X = *(this->xView);
97 const MapType& Y = *(this->yView);
98
99 int i = this->get_index_to_right_of(x);
100
101 // don't extrapolate at all
102 if (i <= 0 || i >= X.size()) return 0;
103
104 // this should be the same t as in the regular interpolation case
105 Real t = (x - X(i - 1)) / (X(i) - X(i - 1));
106
107 Real qprime =
108 (Y(i) - Y(i - 1)) / (X(i) - X(i - 1)) +
109 (1 - 2 * t) * (a[i - 1] * (1 - t) + b[i - 1] * t) / (X(i) - X(i - 1)) +
110 t * (1 - t) * (b[i - 1] - a[i - 1]) / (X(i) - X(i - 1));
111
112 return qprime;
113}

115template <class Real>
116Real CubicSplineInterpolator<Real>::integral(Real _a, Real _b) const {
117 if (this->xView->size() < 1) return 0;
118
119 const MapType& X = *(this->xView);
120 const MapType& Y = *(this->yView);
121
122 // allow b to be less than a
123 int sign = 1;
124 if (_a > _b) {
125 std::swap(_a, _b);
126 sign = -1;
127 }
128 // no extrapolation
129 _a = std::max(_a, X[0]);
130 _b = std::min(_b, X[X.size() - 1]);
131
132 int ai = this->get_index_to_right_of(_a);
133 int bi = this->get_index_to_right_of(_b);
134
169 Real x_1, x_2, t;
170 Real y_1, y_2;
171 Real sum = 0;
172 for (int i = ai; i < bi - 1; i++) {
173 // x_1 -> X(i)
174 // x_2 -> X(i+1)
175 // y_1 -> Y(i)
176 // y_2 -> Y(i+1)
177 x_1 = X[i];
178 x_2 = X[i + 1];
179 y_1 = Y[i];
180 y_2 = Y[i + 1];
181 // X(ai) is to the RIGHT of _a
182 // X(bi) is to the RIGHT of _b, but i only goes up to bi-2 and
183 // X(bi-1) is to the LEFT of _b
184 // therefore, we are just handling interior elements in this loop.
185 sum += static_cast<Real>(
186 (x_2 - x_1) * (0.5 * (y_1 + y_2) + (1. / 12) * (a[i] + b[i])));
187 }
188
189 // now we need to handle the area between [_a,X(ai)] and [X(bi-1),_b]
190
191 // [X(0),_b]
192 // x_1 -> X(bi-1)
193 // x_2 -> X(bi)
194 // y_1 -> Y(bi-1)
195 // y_2 -> Y(bi)
196 x_1 = X[bi - 1];
197 x_2 = X[bi];
198 y_1 = Y[bi - 1];
199 y_2 = Y[bi];
200 t = (_b - x_1) / (x_2 - x_1);
201
202 // adding area between x_1 and _b
203 sum += static_cast<Real>(
204 (x_2 - x_1) *
205 ((t - pow(t, 2) / 2) * y_1 + pow(t, 2) / 2. * y_2 +
206 a[bi - 1] * (pow(t, 2) - 2. * pow(t, 3) / 3. + pow(t, 4) / 4.) +
207 b[bi - 1] * (pow(t, 3) / 3. - pow(t, 4) / 4.)));
208
209 //
210 // [_a,X(0)]
211 // x_1 -> X(ai-1)
212 // x_2 -> X(ai)
213 // y_1 -> Y(ai-1)
214 // y_2 -> Y(ai)
215 x_1 = X[ai - 1];
216 x_2 = X[ai];
217 y_1 = Y[ai - 1];
218 y_2 = Y[ai];
219 t = (_a - x_1) / (x_2 - x_1);
220
221 // subtracting area from x_1 to _a
222 sum -= static_cast<Real>(
223 (x_2 - x_1) *
224 ((t - pow(t, 2) / 2) * y_1 + pow(t, 2) / 2. * y_2 +
225 a[ai - 1] * (pow(t, 2) - 2. * pow(t, 3) / 3. + pow(t, 4) / 4.) +
226 b[ai - 1] * (pow(t, 3) / 3. - pow(t, 4) / 4.)));
227
228 if (ai != bi) // _a and _b are not in the in the same element, need to add
229 // area of element containing _a
230 sum += static_cast<Real>(
231 (x_2 - x_1) *
232 (0.5 * (y_1 + y_2) + (1. / 12) * (a[ai - 1] + b[ai - 1])));
233
234 return sign * sum;
235}

240template <typename Real>
241void CubicSplineInterpolator<Real>::setupInterpolator() {
242 const MapType& X = *(this->xView);
243 const MapType& Y = *(this->yView);
244
245 this->a = VectorType(X.size() - 1);
246 this->b = VectorType(X.size() - 1);
247
248 // we need to solve A x = b, where A is a matrix and b is a vector...
249 // isn't that what we are always doing?
250
251 /*
252 * Solves Ax=B using the Thomas algorithm, because the matrix A will be
253 * tridiagonal and diagonally dominant.
254 *
255 * The method is outlined on the Wikipedia page for Tridiagonal Matrix
256 * Algorithm
257 */
258
259 // init the matrices that get solved
260 VectorType Aa(X.size()), Ab(X.size()), Ac(X.size());
261 VectorType bb(X.size());
262
263 // Ac is a vector of the upper diagonals of matrix A
264 //
265 // Since there is no upper diagonal on the last row, the last value must be
266 // zero.
267 for (int i = 0; i < X.size() - 1; ++i) {
268 Ac(i) = 1 / (X(i + 1) - X(i));
269 }
270
271 // This is the line that was causing breakage for n=odd. It was Ac(X.size())
272 // and should have been Ac(X.size()-1)
273 Ac(X.size() - 1) = 0.0;
274
275 // Ab is a vector of the diagonals of matrix A
276 Ab(0) = 2 / (X(1) - X(0));
277 for (int i = 1; i < X.size() - 1; ++i) {
278 Ab(i) = 2 / (X(i) - X(i - 1)) + 2 / (X(i + 1) - X(i));
279 }
280 Ab(X.size() - 1) = 2 / (X(X.size() - 1) - X(X.size() - 1 - 1));
281
282 // Aa is a vector of the lower diagonals of matrix A
283 //
284 // Since there is no upper diagonal on the first row, the first value must
285 // be zero.
286 Aa(0) = 0.0;
287 for (int i = 1; i < X.size(); ++i) {
288 Aa(i) = 1 / (X(i) - X(i - 1));
289 }
290
291 // setup RHS vector
292 for (int i = 0; i < X.size(); ++i) {
293 if (i == 0) {
294 bb(i) = 3 * (Y(i + 1) - Y(i)) / pow(X(i + 1) - X(i), 2);
295 } else if (i == X.size() - 1) {
296 bb(i) = 3 * (Y(i) - Y(i - 1)) / pow(X(i) - X(i - 1), 2);
297 } else {
298 bb(i) = 3 * ((Y(i) - Y(i - 1)) / (pow(X(i) - X(i - 1), 2)) +
299 (Y(i + 1) - Y(i)) / pow(X(i + 1) - X(i), 2));
300 }
301 }
302
303 VectorType c_star(X.size());
304
305 c_star(0) = Ac(0) / Ab(0);
306 for (int i = 1; i < c_star.size(); ++i) {
307 c_star(i) = Ac(i) / (Ab(i) - Aa(i) * c_star(i - 1));
308 }
309
310 VectorType d_star(X.size());
311 d_star(0) = bb(0) / Ab(0);
312
313 for (int i = 1; i < X.size(); ++i) {
314 d_star(i) =
315 (bb(i) - Aa(i) * d_star(i - 1)) / (Ab(i) - Aa(i) * c_star(i - 1));
316 }
317
318 VectorType x(X.size());
319 x(X.size() - 1) = d_star(X.size() - 1);
320
321 for (int i = X.size() - 1; i-- > 0;) {
322 x(i) = d_star(i) - c_star(i) * x(i + 1);
323 }
324
325 for (int i = 0; i < X.size() - 1; ++i) {
326 this->a(i) = x(i) * (X(i + 1) - X(i)) - (Y(i + 1) - Y(i));
327 this->b(i) = -x(i + 1) * (X(i + 1) - X(i)) + (Y(i + 1) - Y(i));
328 }
329}
