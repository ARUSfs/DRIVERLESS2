116void GraphSlam::state_callback(const common_msgs::msg::State::SharedPtr msg)
117{
118 // Update pose estimation from odometry
119 double phi = vehicle_pose_(2);
120 B.block(0, 0, 2, 2) << std::cos(phi), -std::sin(phi), std::sin(phi), std::cos(phi);
121 u << msg->vx, msg->vy, msg->r;
122 vehicle_pose_ += B * u * dt; // TODO check phi in range [-pi, pi]
123 driven_distance_ += (B * u * dt).norm();
124
125 // Add the odometry to the buffer to re-predict the pose after optimization
126 odom_buffer_.push_back(u);
127
128 if (vehicle_pose_(2) > M_PI) {
129 vehicle_pose_(2) -= 2*M_PI;
130 } else if (vehicle_pose_(2) < -M_PI) {
131 vehicle_pose_(2) += 2*M_PI;
132 }
133
134 prev_t_ = this->now();
135 // Publish the vehicle pose as a tf
136 send_tf();
137 // Check if the vehicle has crossed the finish line
138 check_finish_line();
139
140 // Add a new pose vertex
141 g2o::VertexSE2* pose_vertex = new g2o::VertexSE2();
142 pose_vertex->setId(2*pose_vertices_.size()+1); // Use odd ids for pose vertices
143 pose_vertex->setEstimate(g2o::SE2(vehicle_pose_));
144 pose_vertices_.push_back(pose_vertex);
145 pose_vertices_to_add_.push_back(pose_vertex);
146
147 if (pose_vertices_.size() < 2) {
148 return;
149 }
150
151 if (pose_edges_counter_ % kPoseEdgesSpacing == 0) {
152 // Get the previous pose vertex
153 g2o::VertexSE2* prev_pose_vertex = pose_vertices_[pose_vertices_.size()-2];
154
155 // Add an edge between the previous pose vertex and the new pose vertex
156 g2o::EdgeSE2* pose_edge = new g2o::EdgeSE2();
157 pose_edge->vertices()[0] = prev_pose_vertex;
158 pose_edge->vertices()[1] = pose_vertex;
159 // The weight of the edge is the transform between the two poses
160 g2o::SE2 transform = prev_pose_vertex->estimate().inverse() * pose_vertex->estimate();
161 pose_edge->setMeasurement(transform);
162 // The information matrix is the inverse of the covariance matrix
163 pose_edge->setInformation(Q.inverse());
164 g2o::RobustKernelHuber* rkPose = new g2o::RobustKernelHuber();
165 pose_edge->setRobustKernel(rkPose);
166 rkPose->setDelta(0.3);
167 pose_edges_.push_back(pose_edge);
168 edges_to_add_.push_back(pose_edge);
169
170 pose_edges_counter_ = 0;
171 }
172 pose_edges_counter_++;
173

302void GraphSlam::update_pose_predictions(){
303
304 if(last_optimized_pose_ == nullptr){
305 return;
306 }
307
308 std::vector<Eigen::Vector3d> updated_poses;
309 Eigen::Vector3d updated_pose = last_optimized_pose_->estimate().toVector();
310 updated_poses.push_back(updated_pose);
311
312 int index = last_optimized_pose_->id();
313 Eigen::Matrix3d B2 = Eigen::Matrix3d::Identity();
314 Eigen::Vector3d u2 = Eigen::Vector3d::Zero();
315
316 // Re-predict the pose for each odometry measurement in the buffer
317 for (auto u2: odom_buffer_){
318 double phi = updated_pose(2);
319 B2.block(0, 0, 2, 2) << std::cos(phi), -std::sin(phi), std::sin(phi), std::cos(phi);
320 updated_pose += B2 * u2 * dt;
321
322 updated_poses.push_back(updated_pose);
323 }
324
325 // Update the pose vertices with the new predictions
326 for(auto pose_vertex: pose_vertices_to_add_){
327 pose_vertex->setEstimate(g2o::SE2(updated_poses[(pose_vertex->id() - index)/2]));
328 }
329
330 odom_buffer_.clear();
331
332 // Update the vehicle pose with the last prediction
333 double dx = updated_pose(0)-vehicle_pose_(0);
334 double dy = updated_pose(1)-vehicle_pose_(1);
335 double dyaw = updated_pose(2)-vehicle_pose_(2);
336 if (dyaw > M_PI) {
337 dyaw -= 2*M_PI;
338 } else if (dyaw < -M_PI) {
339 dyaw += 2*M_PI;
340 }
341 double max_offset = 1;
342 double max_yaw_offset = M_PI/18;
343 vehicle_pose_(0) += dx*std::min(1.0, max_offset/std::sqrt(dx*dx+dy*dy));
344 vehicle_pose_(1) += dy*std::min(1.0, max_offset/std::sqrt(dx*dx+dy*dy));
345 vehicle_pose_(2) += dyaw*std::min(1.0, max_yaw_offset/std::abs(dyaw));
346}

349void GraphSlam::fill_graph(){
350 // Get last optimized pose from which predictions will be made
351 last_optimized_pose_ = pose_vertices_.back();
352
353 // Add vertices and edges to the optimizer
354 for (auto vertex: pose_vertices_to_add_){
355 optimizer_.addVertex(vertex);
356 }
357 pose_vertices_to_add_.clear();
358 for (auto vertex: landmark_vertices_to_add_){
359 optimizer_.addVertex(vertex);
360 }
361 landmark_vertices_to_add_.clear();
362 for (auto edge: edges_to_add_){
363 optimizer_.addEdge(edge);
364 }
365 edges_to_add_.clear();
366
367 // Deactivate edges exceding the maximum
368 if(pose_edges_.size()-pose_edges_deactivated_ > kMaxPoseEdges){
369 for (int i=pose_edges_deactivated_; i<pose_edges_.size()-kMaxPoseEdges; i++){
370 pose_edges_[i]->setLevel(2); // Deactivate edge
371 }
372 pose_edges_deactivated_ = pose_edges_.size()-kMaxPoseEdges;
373 // Fix first activated pose, to reduce degrees of freedom
374 g2o::VertexSE2* v0 = static_cast<g2o::VertexSE2*>(pose_edges_[pose_edges_deactivated_]->vertex(0));
375 v0->setFixed(true);
376 }
377 if(landmark_edges_.size()-landmark_edges_deactivated_ > kMaxLandmarkEdges){
378 for (int i=landmark_edges_deactivated_; i<landmark_edges_.size()-kMaxLandmarkEdges; i++){
379 landmark_edges_[i]->setLevel(2); // Deactivate edge
380 }
381 landmark_edges_deactivated_ = landmark_edges_.size()-kMaxLandmarkEdges;
382 }
383}

398
399 std::vector<int> pub_indices;

401 pub_indices.push_back(i);

404 // If clustering is enabled, filter the map to only include the largest cluster
405 if (kMapMaxClusterFilter) {
406 pcl::PointCloud<pcl::PointXYZ>::Ptr map_pcl(new pcl::PointCloud<pcl::PointXYZ>);
407 for (auto landmark : DA.map_){

428 pub_indices = indices.indices;

433 }

440 continue;

458 cone.y = landmark->world_position_(1);

501Eigen::Vector2d GraphSlam::local_to_global(const Eigen::Vector2d& local_pos) {
502 const double phi = vehicle_pose_(2);
503 const double sinphi = std::sin(phi);
504 const double cosphi = std::cos(phi);
505 const Eigen::Matrix2d R = (Eigen::Matrix2d() << cosphi, -sinphi, sinphi, cosphi).finished();
506
507 return R * local_pos + vehicle_pose_.head(2);
508}

514 const double cosphi = std::cos(phi);
