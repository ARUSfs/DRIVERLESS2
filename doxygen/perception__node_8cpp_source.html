83void Perception::lidar_callback(const sensor_msgs::msg::PointCloud2::SharedPtr lidar_msg)
84{

87
88 // Define the necessary variables
89 pcl::PointCloud<pcl::PointXYZI>::Ptr cloud(new pcl::PointCloud<pcl::PointXYZI>);
90 pcl::PointCloud<pcl::PointXYZI>::Ptr cloud_filtered(new pcl::PointCloud<pcl::PointXYZI>);
91 pcl::PointCloud<pcl::PointXYZI>::Ptr cloud_plane(new pcl::PointCloud<pcl::PointXYZI>);

93 std::vector<pcl::PointIndices> cluster_indices;

97
98
99 // Transform the message into a pcl point cloud
100 pcl::fromROSMsg(*lidar_msg, *cloud);
101
102
103 if (kDebug && cloud->size() == 0)
104 {
105 RCLCPP_WARN(this->get_logger(), "Empty point cloud");
106 return;
107 }
108
109
110 if (kCrop)
111 {
112 // Crop the point cloud
113 Cropping::crop_filter_cropbox(cloud, kMaxXFov, kMaxYFov, kMaxZFov);
114 if (kDebug) RCLCPP_INFO(this->get_logger(), "Cropping Time: %f", this->now().seconds() - start_time);
115 }
116
117
118 // Apply the ground filter function
119 GroundFiltering::grid_ground_filter(cloud, cloud_filtered, cloud_plane, coefficients, kThresholdGroundFilter, kMaxXFov, kMaxYFov,
120 kMaxZFov, kNumberSections, kAngleThreshold);
121 if (kDebug) RCLCPP_INFO(this->get_logger(), "Ground Filter Time: %f", this->now().seconds() - start_time);
122
123
124 // Extract the clusters from the point cloud
125 Clustering::euclidean_clustering(cloud_filtered, cluster_indices, kMinClusterSize, kMaxClusterSize);
126 if (kDebug) RCLCPP_INFO(this->get_logger(), "Clustering time: %f", this->now().seconds() - start_time);
127

129 if (kDebug && cluster_indices.empty())

131 RCLCPP_WARN(this->get_logger(), "No clusters found");
132 return;

134
135

138 if (kDebug) RCLCPP_INFO(this->get_logger(), "Center calculation time: %f", this->now().seconds() - start_time);
139
140
141 // Recover ground points
142 Utils::reconstruction(cloud_plane, cloud_filtered, cluster_indices, clusters_centers, kRadius);
143 if (kDebug) RCLCPP_INFO(this->get_logger(), "Reconstruction time: %f", this->now().seconds() - start_time);
144
145
146 // Filter the clusters by size
147 Utils::filter_clusters(cluster_indices, cloud_filtered, clusters_centers);
148 if (kDebug) RCLCPP_INFO(this->get_logger(), "Filtering time: %f", this->now().seconds() - start_time);
149
150
151 // Convert the indices of the clusters to the points of the clusters
152 std::vector<pcl::PointCloud<pcl::PointXYZI>::Ptr> cluster_points;
153 for (const auto& cluster : cluster_indices)
154 {
155 pcl::PointCloud<pcl::PointXYZI>::Ptr new_cluster(new pcl::PointCloud<pcl::PointXYZI>);
156 for (const auto& idx : cluster.indices)
157 {
158 pcl::PointXYZI point = cloud_filtered->points[idx];
159 new_cluster->points.push_back(point);
160 }
161 cluster_points.push_back(new_cluster);
162 }
163
164
165 if (kDebug)
166 {
167 // Publish clusters cloud with different colors
168 int i = 0;
169 for (auto c : cluster_points)
170 {
171 for (auto &p : c->points)
172 {
173 pcl::PointXYZI q = p;
174 q.intensity = i;
175 clusters_cloud->push_back(q);
176 }
177 i++;
178 }
179 }
180
181
182 // Score the clusters and keep the ones that will be considered cones
183 Scoring::scoring_surface(final_map, cluster_points, clusters_centers, kThresholdScoring);
184 if (kDebug) RCLCPP_INFO(this->get_logger(), "Scoring time: %f", this->now().seconds() - start_time);
185
186

188 {
189 RCLCPP_WARN(this->get_logger(), "No cones found");
190 return;
191 }
192
193
194 // Estimate the color of the closest cones
195 if (kColor)
196 {
197 ColorEstimation::color_estimation(cluster_points, clusters_centers, kDistanceThreshold, kColoringThreshold);
198 if (kDebug) RCLCPP_INFO(this->get_logger(), "Color estimation time: %f", this->now().seconds() - start_time);
199
200
201 // Update the colors of final map points
202 for (auto& point : final_map->points)
203 {
204 for (const auto& center : clusters_centers)
205 {
206 if (point.x == center.x && point.y == center.y && point.z == center.z)
207 {
208 point.color = center.color;
209 }
210 }
211 }
212 }
213
214
215 // Motion correction
216 double dt = this->now().seconds() - start_time; // Estimate SDK delay
217 Utils::motion_correction(final_map, vx, vy, yaw_rate, dt);
218 if (kDebug) RCLCPP_INFO(this->get_logger(), "Motion correction time: %f", this->now().seconds() - start_time);
219
220
221 final_times.push_back(this->now().seconds() - start_time);
222 double average_time = std::accumulate(final_times.begin(), final_times.end(), 0.0) / final_times.size();
223 if (kDebug && final_map->size() > 0) RCLCPP_INFO(this->get_logger(), "Number of cones: %zu", final_map->size());
224 if (kDebug) RCLCPP_INFO(this->get_logger(), "Final iteration time: %f", this->now().seconds() - start_time);
225 if (kDebug) RCLCPP_INFO(this->get_logger(), "Average time: %f", average_time);
226 if (kDebug) RCLCPP_INFO(this->get_logger(), "//////////////////////////////////////////////");

228
229 if (kDebug)
230 {
231 // Publish the filtered cloud
232 sensor_msgs::msg::PointCloud2 filtered_msg;
233 pcl::toROSMsg(*cloud_filtered, filtered_msg);
234 filtered_msg.header.frame_id = "/rslidar";
235 filtered_pub_->publish(filtered_msg);

241 clusters_pub_->publish(clusters_msg);

250}
