116void Perception::lidar_callback(const sensor_msgs::msg::PointCloud2::SharedPtr lidar_msg)
117{
118 double start_time = this->now().seconds();
119
120
121 // Transform the message into a pcl point cloud
122 pcl::PointCloud<pcl::PointXYZI>::Ptr cloud(new pcl::PointCloud<pcl::PointXYZI>);
123 pcl::fromROSMsg(*lidar_msg, *cloud);
124 Eigen::Matrix4f T_lidar_to_cog = Eigen::Matrix4f::Identity();
125 T_lidar_to_cog(0, 3) = kLidarCogX; // x translation
126 pcl::transformPointCloud(*cloud, *cloud, T_lidar_to_cog);
127
128 if (cloud->size() == 0)
129 {
130 if (kDebug) RCLCPP_WARN(this->get_logger(), "Empty point cloud");
131 return;
132 }
133
134
135 if (kAccumBufferSize > 0) {
136 this->accumulate(cloud);
137 if (kDebug) {
138 // Publish the accumulation cloud
139 sensor_msgs::msg::PointCloud2 acum_msg;
140 pcl::toROSMsg(*acum_cloud_, acum_msg);
141 acum_msg.header.frame_id = "/rslidar";
142 accumulation_pub_->publish(acum_msg);
143
144 RCLCPP_INFO(this->get_logger(), "Accumulation time: %f", this->now().seconds() - start_time);
145 }
146
147 cloud = acum_cloud_; // Use the accumulated cloud for processing
148 }
149
150 if (kCrop)
151 {
152 // Crop the point cloud
153 double min_x = 0.0;
154 double min_y = -kMaxYFov;
155 double min_z = -1.0;
156 Cropping::box_filter<pcl::PointXYZI>(cloud, cloud, min_x, min_y, min_z, kMaxXFov, kMaxYFov, kMaxZFov);
157 if (kDebug) RCLCPP_INFO(this->get_logger(), "Cropping Time: %f", this->now().seconds() - start_time);
158 }
159
160
161 // Apply the ground filter function
162 pcl::PointCloud<pcl::PointXYZI>::Ptr cloud_filtered(new pcl::PointCloud<pcl::PointXYZI>);
163 pcl::PointCloud<pcl::PointXYZI>::Ptr ground_cloud(new pcl::PointCloud<pcl::PointXYZI>);
164
165 if (kGroundFilterType == "Z_FILTER") {
166 Cropping::axis_filter<pcl::PointXYZI>(cloud, cloud_filtered, "z", 0.08, 0.4);
167 } else if (kGroundFilterType == "RANSAC") {
168 pcl::ModelCoefficients::Ptr coefficients(new pcl::ModelCoefficients);
169 GroundFiltering::ransac_ground_filter(cloud, cloud_filtered, ground_cloud, coefficients, kThresholdGroundFilter);
170 } else if (kGroundFilterType == "RANSAC_GRID") {
171 pcl::ModelCoefficients::Ptr coefficients(new pcl::ModelCoefficients);
172 GroundFiltering::grid_ground_filter(cloud, cloud_filtered, ground_cloud, coefficients, kThresholdGroundFilter,
173 kMaxXFov, kMaxYFov, kMaxZFov, kNumberSections, kAngleThreshold);
174 } else if (kGroundFilterType == "PILLAR") {
175 GroundFiltering::pillar_ground_filter(cloud, cloud_filtered, ground_cloud, kThresholdGroundFilter,
176 kMaxXFov, kMaxYFov, kNumberSections);
177 } else {
178 RCLCPP_ERROR(this->get_logger(), "Unknown ground filter type: %s", kGroundFilterType.c_str());
179 return;
180 }
181 if (kDebug) RCLCPP_INFO(this->get_logger(), "Ground Filter Time: %f", this->now().seconds() - start_time);
182
183
184 // Extract the clusters from the point cloud
185 std::vector<pcl::PointCloud<pcl::PointXYZI>::Ptr> cluster_clouds;
186 Clustering::FEC<pcl::PointXYZI>(cloud_filtered, &cluster_clouds, kTolerance, kMinClusterSize + 1*kAccumBufferSize, kMaxClusterSize + 100*kAccumBufferSize);
187 if (kDebug) RCLCPP_INFO(this->get_logger(), "Clustering time: %f", this->now().seconds() - start_time);
188
189
190 if (kDebug && cluster_clouds.empty())
191 {
192 RCLCPP_WARN(this->get_logger(), "No clusters found");
193 return;
194 }
195
196 std::vector<pcl::PointXYZI> clusters_centers;
197 Utils::get_clusters_centers(&cluster_clouds, &clusters_centers);
198
199
200 if (kReconstruction && !ground_cloud->empty())
201 {
202 // Filter too big clusters
203 Utils::filter_clusters(&cluster_clouds, &clusters_centers, 0.0, 0.4, 0.5, 1.0);
204
205 // Recover ground points
206 Utils::reconstruction(ground_cloud, &cluster_clouds, &clusters_centers, kRecRadius);
207 if (kDebug) RCLCPP_INFO(this->get_logger(), "Reconstruction time: %f", this->now().seconds() - start_time);
208 }
209
210 // Filter too big and too small clusters
211 Utils::filter_clusters(&cluster_clouds, &clusters_centers);
212
213
214 // Score the clusters and keep the ones that will be considered cones
215 pcl::PointCloud<PointXYZProbColorScore>::Ptr final_map(new pcl::PointCloud<PointXYZProbColorScore>);
216 Scoring::scoring_surface(final_map, cluster_clouds, clusters_centers, kThresholdScoring);
217 if (kDebug) RCLCPP_INFO(this->get_logger(), "Scoring time: %f", this->now().seconds() - start_time);
218
219
220 if (kDebug && final_map->size() == 0)
221 {
222 RCLCPP_WARN(this->get_logger(), "No cones found");
223 return;
224 }
225
226
227 // Estimate the color of the closest cones
228 if (kColor)
229 {
230 Coloring::color_estimation(final_map, cluster_clouds, clusters_centers, kDistanceThreshold, kDebug);
231 if (kDebug) RCLCPP_INFO(this->get_logger(), "Color estimation time: %f", this->now().seconds() - start_time);
232 }
233
234
235 // Motion correction
236 double dt = this->now().seconds() - start_time; // Estimate SDK delay
237 // Remove or adapt this if Utils::motion_correction expects a different point type
238 // Utils::motion_correction(final_map, vx_, vy_, r_, dt);
239 if (kDebug) RCLCPP_INFO(this->get_logger(), "Motion correction time: %f", this->now().seconds() - start_time);
240
241
242 final_times.push_back(this->now().seconds() - start_time);
243 double average_time = std::accumulate(final_times.begin(), final_times.end(), 0.0) / final_times.size();
244 if (kDebug && final_map->size() > 0) RCLCPP_INFO(this->get_logger(), "Number of cones: %zu", final_map->size());
245 if (kDebug) RCLCPP_INFO(this->get_logger(), "Final iteration time: %f", this->now().seconds() - start_time);
246 if (kDebug) RCLCPP_INFO(this->get_logger(), "Average time: %f", average_time);
247 if (kDebug) RCLCPP_INFO(this->get_logger(), "//////////////////////////////////////////////");
248
249
250 if (kDebug)
251 {
252 // Publish the filtered cloud
253 sensor_msgs::msg::PointCloud2 filtered_msg;
254 pcl::toROSMsg(*cloud_filtered, filtered_msg);
255 filtered_msg.header.frame_id = "/rslidar";
256 filtered_pub_->publish(filtered_msg);
257
258 // Publish the clusters cloud
259 pcl::PointCloud<pcl::PointXYZI>::Ptr all_clusters_cloud(new pcl::PointCloud<pcl::PointXYZI>);
260 for (const auto& c : cluster_clouds) *all_clusters_cloud += *c;
261 sensor_msgs::msg::PointCloud2 clusters_msg;
262 pcl::toROSMsg(*all_clusters_cloud, clusters_msg);
263 clusters_msg.header.frame_id = "/rslidar";
264 clusters_pub_->publish(clusters_msg);
265 }
266
267
268 // Publish the map cloud
269 sensor_msgs::msg::PointCloud2 map_msg;
270 pcl::toROSMsg(*final_map, map_msg);
271 map_msg.header.frame_id = "/rslidar";
272 map_pub_->publish(map_msg);
273}

277void Perception::accumulate(pcl::PointCloud<pcl::PointXYZI>::Ptr& cloud){
278 // Get ground plane with RANSAC and align the cloud
279 pcl::PointCloud<pcl::PointXYZI>::Ptr actual_cloud_filtered(new pcl::PointCloud<pcl::PointXYZI>);
280 pcl::PointCloud<pcl::PointXYZI>::Ptr actual_ground_cloud(new pcl::PointCloud<pcl::PointXYZI>);
281 pcl::ModelCoefficients::Ptr coefficients(new pcl::ModelCoefficients);
282 GroundFiltering::ransac_ground_filter(cloud, actual_cloud_filtered, actual_ground_cloud, coefficients, 0.05);
283 Utils::ground_align(cloud, coefficients);
284
285 // Filter the ground points
286 Cropping::axis_filter<pcl::PointXYZI>(cloud, cloud, "z", 0.08, 1.0);
287
288 // Extract clusters estimation
289 std::vector<pcl::PointCloud<pcl::PointXYZI>::Ptr> cluster_clouds;
290 Clustering::FEC<pcl::PointXYZI>(cloud, &cluster_clouds, kTolerance, kMinClusterSize, kMaxClusterSize);
291
292 std::vector<pcl::PointXYZI> clusters_centers;
293 Utils::get_clusters_centers(&cluster_clouds, &clusters_centers);
294 Utils::filter_clusters(&cluster_clouds, &clusters_centers);
295
296 // Get cones estimation
297 pcl::PointCloud<pcl::PointXYZ>::Ptr actual_cones(new pcl::PointCloud<pcl::PointXYZ>);
298 for (auto center : clusters_centers) actual_cones->points.push_back(pcl::PointXYZ(center.x, center.y, center.z));
299
300 if (cloud_buffer_.size() == 0)
301 {
302 cloud_buffer_.push_back(cloud);
303 *prev_cones_ = *actual_cones;
304 return;
305 }
306
307
308 // Update previous cones estimation
309 Eigen::Matrix4f transform = Eigen::Matrix4f::Identity();
310 transform(0, 3) = -vx_ * dt; // x translation
311 transform(1, 3) = -vy_ * dt; // y translation
312 transform(2, 3) = 0.0; // z translation (assuming no vertical movement)
313 double phi = - r_ * dt; // rotation around z-axis
314 transform(0, 0) = cos(phi);
315 transform(0, 1) = -sin(phi);
316 transform(1, 0) = sin(phi);
317 transform(1, 1) = cos(phi);
318 pcl::transformPointCloud(*prev_cones_, *prev_cones_, transform);
319
320
321 // Pair the previous cones with the actual cones using a KdTree
322 pcl::KdTreeFLANN<pcl::PointXYZ> kdtree;
323 kdtree.setInputCloud(actual_cones);
324
325 pcl::PointCloud<pcl::PointXYZ>::Ptr paired_src(new pcl::PointCloud<pcl::PointXYZ>);
326 pcl::PointCloud<pcl::PointXYZ>::Ptr paired_dst(new pcl::PointCloud<pcl::PointXYZ>);
327
328 for (auto& pt : prev_cones_->points) {
329 std::vector<int> indices(1);
330 std::vector<float> sqr_dists(1);
331
332 if (kdtree.nearestKSearch(pt, 1, indices, sqr_dists) > 0) {
333 if (sqr_dists[0] <= 1.0f) { // 1 metro al cuadrado
334 pt.z = 0;
335 pcl::PointXYZ pt_dest = actual_cones->points[indices[0]];
336 pt_dest.z = 0; // Height at 0 to ensure the transformation is in the XY plane
337 paired_src->points.push_back(pt);
338 paired_dst->points.push_back(pt_dest);
339 }
340 }
341 }
342
343 *prev_cones_ = *actual_cones;
344
345 // Estimate the transformation between the two point clouds with SVD
346 Eigen::Matrix4f transform2;
347 pcl::registration::TransformationEstimationSVD<pcl::PointXYZ, pcl::PointXYZ> svd;
348 svd.estimateRigidTransformation(*paired_src, *paired_dst, transform2);
349
350
351 // Accumulate the point cloud using motion and SVD transformations
352 acum_cloud_->clear();
353 for (auto& pcl : cloud_buffer_)
354 {
355 pcl::PointCloud<PointXYZProbColorScore>::Ptr cones_map(new pcl::PointCloud<PointXYZProbColorScore>);
356 pcl::transformPointCloud(*pcl, *pcl, transform);
357 pcl::transformPointCloud(*pcl, *pcl, transform2);
358 *acum_cloud_ += *pcl;
359 }
360
361
362 if (cloud_buffer_.size() > kAccumBufferSize)
363 {
364 cloud_buffer_.pop_back();
365 }
366 cloud_buffer_.insert(cloud_buffer_.begin(), cloud);
367
368 if (kVoxelFilter) {
369 pcl::VoxelGrid<pcl::PointXYZI> voxel_filter;
370 voxel_filter.setInputCloud(acum_cloud_);
371 voxel_filter.setLeafSize(kVoxelSizeX, kVoxelSizeY, kVoxelSizeZ);
372 voxel_filter.filter(*acum_cloud_);
373 }
374
375}
