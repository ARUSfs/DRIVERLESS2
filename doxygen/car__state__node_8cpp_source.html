269void CarState::on_timer()
270{
271 if (kMission!="inspection"){
272 this->get_tf_position();
273 }
274
275 // Estimate vx, vy
276 double avg_vx;
277 if(kSimulation){
278
279 double sin = std::sin(delta_);
280 double cos = std::cos(delta_);
281
282 double vx_fl_cog = (v_front_left_ - sin*vy_ -(kLf*sin - 0.5*kTf*cos)*r_) / cos;
283 double vx_fr_cog = (v_front_right_ - sin*vy_ -(kLf*sin + 0.5*kTf*cos)*r_) / cos;
284 double vx_rl_cog = v_rear_left_ + 0.5*kTr*r_;
285 double vx_rr_cog = v_rear_right_ - 0.5*kTr*r_;
286
287 avg_vx = (vx_fl_cog+vx_fr_cog+vx_rl_cog+vx_rr_cog)/4;
288
289 } else if (!kSimulation && kUseWheelspeeds
290 && inv_speed_ > 2.0 && std::abs(v_front_right_) < 1.5*kMaxVx && std::abs(v_rear_right_) < 1.5*kMaxVx) {
291
292 double sin = std::sin(delta_);
293 double cos = std::cos(delta_);
294
295 // double vx_fl_cog = (v_front_left_ - sin*vy_ -(kLf*sin - 0.5*kTf*cos)*r_) / cos;
296 double vx_fr_cog = (v_front_right_ - sin*vy_ -(kLf*sin + 0.5*kTf*cos)*r_) / cos;
297 // double vx_rl_cog = v_rear_left_ + 0.5*kTr*r_;
298 double vx_rr_cog = v_rear_right_ - 0.5*kTr*r_;
299
300 avg_vx = (vx_fr_cog + vx_rr_cog)/2;
301
302 } else {
303
304 avg_vx = inv_speed_;
305
306 }
307
308 Vector2d x_est = speed_estimator_.estimate_speed(ax_, r_, delta_, delta_der_, avg_vx);
309 vx_ = x_est(0);
310 vy_ = x_est(1);
311
312 // Publish state message
313 auto state_msg = common_msgs::msg::State();
314
315 state_msg.header.stamp = this->now();
316 state_msg.x = x_;
317 state_msg.y = y_;
318 state_msg.yaw = yaw_;
319 state_msg.vx = vx_;
320 state_msg.vy = vy_;
321 state_msg.r = r_;
322 state_msg.ax = ax_;
323 state_msg.ay = ay_;
324 state_msg.delta = delta_;
325
326 state_pub_->publish(state_msg);
327
328 // Check if the mission is finished
329 if(as_status_== 3 && target_speed_== 0.0 && vx_ < 2.5){
330 as_status_ = 5;
331 }
332
333 // Publish car info message
334 auto car_info_msg = common_msgs::msg::CarInfo();
335
336 car_info_msg.header.stamp = this->now();
337 car_info_msg.as_status = as_status_;
338 car_info_msg.ami = ami_;
339 car_info_msg.ebs_status = ebs_status_;
340 car_info_msg.ebs_redundancy_status = ebs_redundancy_status_;
341 car_info_msg.x = x_;
342 car_info_msg.y = y_;
343 car_info_msg.yaw = yaw_;
344 car_info_msg.vx = vx_;
345 car_info_msg.vy = vy_;
346 car_info_msg.target_speed = target_speed_;
347 car_info_msg.r = r_;
348 car_info_msg.ax = ax_;
349 car_info_msg.ay = ay_;
350 car_info_msg.delta = delta_;
351 car_info_msg.target_delta = target_delta_;
352 car_info_msg.steering_state = steering_state_;
353 car_info_msg.brake_hydr_pressure = brake_hydr_pressure_;
354 car_info_msg.lap_count = lap_count_;
355 car_info_msg.cones_count_actual = cones_count_actual_;
356 car_info_msg.cones_count_all = cones_count_all_;
357
358 car_info_pub_->publish(car_info_msg);
359
360
361 // Decrease plausability buffer
362 if (plausability_ > 0){
363 plausability_--;
364 }
365
366
367 // Publish run check
368 auto run_check_msg = std_msgs::msg::Bool();
369 if (kSafeMode){
370 run_check_msg.data = (as_status_ == 3) && (plausability_ < kMaxPlausabilityError) && epos_OK_;
371 } else {
372 run_check_msg.data = (as_status_ == 3);
373 }
374
375 run_check_pub_->publish(run_check_msg);
376
377 // Publish steering check
378 auto steering_check_msg = std_msgs::msg::Bool();
379 steering_check_msg.data = run_check_msg.data && (vx_ >= 0.5);
380 steer_check_pub_->publish(steering_check_msg);
381
382}

401{
402 auto now_time = this->now();

657void CarState::get_tf_position()
658{
659 geometry_msgs::msg::TransformStamped transform;
660 try {
661 transform = tf_buffer_->lookupTransform("arussim/world", "slam/vehicle", tf2::TimePointZero);
662 tf2::Quaternion q(
663 transform.transform.rotation.x,
664 transform.transform.rotation.y,
665 transform.transform.rotation.z,
666 transform.transform.rotation.w
667 );
668 double roll, pitch, yaw;
669 tf2::Matrix3x3(q).getRPY(roll, pitch, yaw);
670 x_ = transform.transform.translation.x;
671 y_ = transform.transform.translation.y;
672 yaw_ = yaw;
673 } catch (tf2::TransformException &ex) {
674 if(kDebug) RCLCPP_WARN(this->get_logger(), "Transform not available: %s", ex.what());
675 }
