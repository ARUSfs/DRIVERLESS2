69void AccPlanning::compute_profiles() {
70 s_.clear();
71 speed_profile_.clear();
72 acc_profile_.clear();
73
74 for (int i = 0; i < 2 * kTrackLength / kStep; i++) {
75 s_.push_back(i * kStep);
76 speed_profile_.push_back(0.0);
77 }
78
79 speed_profile_[0] = 1;
80
81 for (size_t i = 1; i < s_.size(); ++i) {
82 double ds = kStep;
83 if (s_[i] < kTrackLength) {
84 speed_profile_[i] = std::min(kTargetSpeed, sqrt(pow(speed_profile_[i - 1], 2) + 2 * kMaxXAcc * ds));
85 } else {
86 speed_profile_[i] = std::max(0.0, sqrt(pow(speed_profile_[i - 1], 2) - 2 * kMaxDec * ds));
87 }
88 }
89
90 for (size_t i = 0; i < speed_profile_.size() - 1; ++i) {
91 double ds = kStep;
92 acc_profile_.push_back((pow(speed_profile_[i + 1], 2) - pow(speed_profile_[i], 2)) / (2 * ds));
93 }
94}

97void AccPlanning::generate_planning() {
98 std::random_device rd;
99 std::mt19937 gen(rd());
100
101 double a_initial = 0.0;
102 double b_initial = 0.0;
103
104 double best_a1 = 0.0, best_b1 = 0.0;
105 double best_a2 = 0.0, best_b2 = 0.0;
106
107 double max_score = 0.0;
108
109 std::uniform_int_distribution<> distrib(0, cones_.points.size() - 1);
110
111 std::vector<int> inliers_indices1;
112
113 for (int iter = 0; iter < kMaxIterations; ++iter) {
114 int i = distrib(gen);
115 int j = distrib(gen);
116 auto& c1 = cones_.points[i];
117 auto& c2 = cones_.points[j];
118
119 if (std::abs(c1.x - c2.x) < 0.1) {
120 continue;
121 }
122
123 double a1 = (c2.y - c1.y) / (c2.x - c1.x);
124 double b1_1 = c1.y - c1.x * a1;
125
126 double score = 0.0;
127 std::vector<int> temp_inliers_indices;
128
129 for (int idx = 0; idx < cones_.points.size(); ++idx) {
130 const auto& cone = cones_.points[idx];
131 double d = std::abs(a1 * cone.x + b1_1 - cone.y) / std::sqrt(a1 * a1 + 1);
132
133 double weight = 0.25;
134 if (cone.x >= 0.0) {
135 weight = std::exp(-cone.x / 6.0);
136 }
137
138 if (d < kRansacThreshold) {
139 score += weight * (1.0 - (d / kRansacThreshold));
140 temp_inliers_indices.push_back(idx);
141 }
142 }
143
144 if (score > max_score) {
145 best_a1 = a1;
146 best_b1 = b1_1;
147 max_score = score;
148 inliers_indices1 = temp_inliers_indices;
149 }
150 }
151 a_initial = best_a1;
152 b_initial = best_b1;
153
154 pcl::PointCloud<ConeXYZColorScore> remaining_cones;
155 for (int idx = 0; idx < cones_.points.size(); ++idx) {
156 if (std::find(inliers_indices1.begin(), inliers_indices1.end(), idx) == inliers_indices1.end()) {
157 remaining_cones.push_back(cones_.points[idx]);
158 }
159 }
160
161 if (remaining_cones.size() < kMinInliersRequired) {
162 if (kDebug) RCLCPP_WARN(this->get_logger(), "Not enough remaining cones for second line fitting.");
163 return;
164 }
165
166 max_score = 0;
167 std::uniform_int_distribution<> distrib_remaining(0, remaining_cones.size() - 1);
168
169 for (int iter2 = 0; iter2 < kMaxIterations; ++iter2) {
170 int i = distrib_remaining(gen), j = distrib_remaining(gen);
171 const auto& c1 = remaining_cones.points[i];
172 const auto& c2 = remaining_cones.points[j];
173
174 if (std::abs(c1.x - c2.x) < 0.1) {
175 continue;
176 }
177
178 double a2 = (c2.y - c1.y) / (c2.x - c1.x);
179 double b2_1 = c1.y - c1.x * a2;
180
181 double score = 0.0;
182 for (const auto& cone : remaining_cones.points) {
183 double d = std::abs(a2 * cone.x + b2_1 - cone.y) / std::sqrt(a2 * a2 + 1);
184
185 double weight = 0.25;
186 if (cone.x >= 0.0) {
187 weight = std::exp(-cone.x / 6.0);
188 }
189
190 if (d < kRansacThreshold) {
191 score += weight * (1.0 - (d / kRansacThreshold));
192 }
193 }
194
195 if (score > max_score) {
196 best_a2 = a2;
197 best_b2 = b2_1;
198 max_score = score;
199 }
200 }
201
202 double a_temp = (best_a1 + best_a2) / 2;
203 double b_temp = (best_b1 + best_b2) / 2;
204 if (std::abs(a_temp - a_initial) > kMaxAllowedDeviation || std::abs(b_temp - b_initial) > kMaxAllowedDeviation) {
205 if (kDebug) RCLCPP_WARN(this->get_logger(), "Deviation too high. Keeping initial trajectory.");
206 return;
207 }
208
209 a_ = a_temp;
210 b_ = b_temp;
211
212
213 if (kDebug) RCLCPP_INFO(this->get_logger(), "Acceleration trajectory coefficients: a = %f, b = %f", a_, b_);
214}

217void AccPlanning::publish_trajectory() {
218 common_msgs::msg::Trajectory trajectory_msg;
219
220 for (size_t i = 0; i < s_.size(); ++i) {
221 common_msgs::msg::PointXY point;
222 point.x = i * kStep;
223 point.y = a_ * point.x + b_; // Updates the slope of the trajectory
224 trajectory_msg.points.push_back(point);

226 trajectory_msg.speed_profile.push_back(speed_profile_[i]);
