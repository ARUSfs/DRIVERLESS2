68void MonotonicInterpolator<Real>::setupInterpolator() {
69 const MapType& X = *(this->xView);
70 const MapType& Y = *(this->yView);
71
72 a = VectorType(X.size() - 1);
73 b = VectorType(X.size() - 1);
74 yplow = VectorType(X.size() - 1);
75 yphigh = VectorType(X.size() - 1);
76
77 for (int i = 0; i < X.size() - 1; i++) {
78 // i is the *interval* index
79 // the interval spans from X(i) to X(i+1)
80 Real xlow = X(i);
81 Real xhigh = X(i + 1);
82 Real ylow = Y(i);
83 Real yhigh = Y(i + 1);
84 Real h = xhigh - xlow;
85
86 Real slope = (yhigh - ylow) / h;
87
88 // Determine the first derivative values used at the endpoints of the
89 // interval These values may be limited to preserve monotonicity
90 if (i == 0) {
91 // first interval does not have an interval to its "left", so just
92 // use the slope in the interval.
93 yplow[i] = slope;
94 } else {
95 // Determine the lower slope value
96 Real hlow = xlow - X(i - 1);
97 Real slope_low = 0.0;
98 if (hlow > 0.0) slope_low = (ylow - Y(i - 1)) / hlow;
99 if (slope_low * slope <= 0.0) {
100 // Set derivative as zero
101 yplow[i] = 0.0;
102 } else {
103 yplow[i] = ((slope_low * h) + (slope * hlow)) / (hlow + h);
104 if (yplow[i] >= 0.0) {
105 yplow[i] =
106 (std::min)(yplow[i], static_cast<Real>(2.0) *
107 (std::min)(slope_low, slope));
108 } else {
109 yplow[i] =
110 (std::max)(yplow[i], static_cast<Real>(2.0) *
111 (std::max)(slope_low, slope));
112 }
113 }
114 }
115
116 if (i == (X.size() - 2)) {
117 // last interval does not have an interval to its "right", so just
118 // use the slope in the interval.
119 yphigh[i] = slope;
120 } else {
121 // Determine the upper slope value
122 Real hhigh = X(i + 2) - xhigh;
123 Real slope_high = 0.0;
124 if (hhigh > 0.0) slope_high = (Y(i + 2) - yhigh) / hhigh;
125 if (slope * slope_high <= 0.0) {
126 // Set derivative as zero
127 yphigh[i] = 0.0;
128 } else {
129 yphigh[i] = ((slope * hhigh) + (slope_high * h)) / (h + hhigh);
130 if (yphigh[i] >= 0.0) {
131 yphigh[i] = (std::min)(
132 yphigh[i],
133 static_cast<Real>(2.0) * (std::min)(slope, slope_high));
134 } else {
135 yphigh[i] = (std::max)(
136 yphigh[i],
137 static_cast<Real>(2.0) * (std::max)(slope, slope_high));
138 }
139 }
140 }
141
142 a[i] = (yplow[i] + yphigh[i] - (2.0 * slope)) / (h * h);
143 b[i] = ((3.0 * slope) + (-2.0 * yplow[i]) - yphigh[i]) / h;
144 }
145}

147template <class Real>
148Real MonotonicInterpolator<Real>::operator()(Real x) const {
149 BASE::checkData();
150
151 const MapType& X = *(this->xView);
152 const MapType& Y = *(this->yView);
153
154 // don't extrapolate at all
155 if (x < X(0) || x > X(X.size() - 1)) return 0;
156
157 // the index that is just to the left of x will correspond to the "interval
158 // index"
159 int i = this->get_index_to_left_of(x);
160
161 // Deal with the degenerate case of xval = xlow = xhigh
162 if (X(i + 1) <= X(i)) return 0.5 * (Y(i) + Y(i + 1));
163
164 Real xm = x - X(i);
165 Real xm2 = xm * xm;
166 Real f = (a[i] * xm * xm2) + (b[i] * xm2) + (yplow[i] * xm) + Y(i);
167
168 return f;
169}
