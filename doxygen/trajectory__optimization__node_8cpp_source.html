51void TrajectoryOptimization::trajectory_callback(common_msgs::msg::TrackLimits::SharedPtr track_limits_msg){
52 // Extract track limits and trajectory from trajectory message
53 track_limit_right_ = track_limits_msg->right_limit;
54 track_limit_left_ = track_limits_msg->left_limit;
55
56 common_msgs::msg::Trajectory trajectory = track_limits_msg-> trajectory;
57 std::vector<common_msgs::msg::PointXY> track_xy = trajectory.points;
58
59 //Convert trajectory message to vectors
60 int n = track_xy.size();
61 VectorXd x(n), y(n);
62 for(int i = 0; i < n; i++){
63 x(i) = track_xy[i].x;
64 y(i) = track_xy[i].y;
65 }
66
67 if(!(track_limit_left_.empty())){
68 //Generate track width vectors
69 VectorXd twr = TrajectoryOptimization::generate_track_width(x, y, track_limit_right_);
70 VectorXd twl = TrajectoryOptimization::generate_track_width(x, y, track_limit_left_);
71
72 //Get minimal curvature path iteratively
73 MatrixXd optimized_trajectory(n,2);
74
75 for(int i=0; i<kNIter; i++){
76 double t0 = this->now().seconds();
77 auto opt_out = MinCurvaturepath::get_min_curvature_path(x, y, twr, twl, kNSeg);
78 optimized_trajectory = opt_out.first;
79 auto status = opt_out.second;
80
81 if(status != qpmad::Solver::OK){
82 if (kDebug) RCLCPP_ERROR(this->get_logger(), "qpmad solver didn't find a solution at iteration %d!", i+1);
83 } else {
84 x = optimized_trajectory.col(0);
85 y = optimized_trajectory.col(1);
86
87 twr = TrajectoryOptimization::generate_track_width(x, y, track_limit_right_);
88 twl = TrajectoryOptimization::generate_track_width(x, y, track_limit_left_);
89
90 if (kDebug) {
91 RCLCPP_INFO(this->get_logger(), "Trajectory correctly optimized at iteration %d.", i+1);
92 RCLCPP_INFO(this->get_logger(), "Trajectory optimization time: %f", this->now().seconds()-t0);
93 }
94 }
95
96 }
97
98 //Get accumulated distance and curvature at each point
99 MatrixXd optimized_s_k = TrajectoryOptimization::get_distance_and_curvature_values(x, y);
100 VectorXd optimized_s = optimized_s_k.col(0);
101 VectorXd optimized_k = optimized_s_k.col(1);
102
103 //Generate speed and acceletation profile
104 MatrixXd profile = TrajectoryOptimization::generate_speed_and_acc_profile(optimized_s, optimized_k);
105 VectorXd speed_profile = profile.col(0);
106 VectorXd acc_profile = profile.col(1);
107
108 //Create and publish trajectory message
109 common_msgs::msg::Trajectory optimized_traj_msg = TrajectoryOptimization::create_trajectory_msg(x, y, optimized_s, optimized_k, speed_profile, acc_profile);
110 optimized_trajectory_pub_ -> publish(optimized_traj_msg);
111
112 if (kDebug) RCLCPP_INFO(this->get_logger(), "Optimized trajectory correctly published. \n");
113
114 } else {
115 if (kDebug) RCLCPP_ERROR(this->get_logger(), "Track limits empty! \n");
116 }

120VectorXd TrajectoryOptimization::generate_track_width(VectorXd x, VectorXd y, std::vector<common_msgs::msg::PointXY> track_limit){
121 int n = x.size();
122 int m = track_limit.size();
123 VectorXd track_width = VectorXd::Zero(n);
124
125 for(int i = 0; i < n; i++){
126 // Calculate point's distance² to track limits
127 double min_dist = 100;
128 for(int j = 0; j < m; j++){
129 double dx = x(i) - track_limit[j].x;
130 double dy = y(i) - track_limit[j].y;
131 double dist_iter = dx*dx + dy*dy;
132 if (dist_iter < min_dist){min_dist = dist_iter;}
133 }
134 // Ensure minimal distance
135 track_width(i) = std::max(std::sqrt(min_dist) - kMinDist, 0.);
136 }
137
138 return track_width;
139}

141MatrixXd TrajectoryOptimization::get_distance_and_curvature_values(VectorXd traj_x, VectorXd traj_y){
142 // Get the accumulated distance at each point of the trajectory (s)
143 double acum = 0;
144 int n = traj_x.size();
145 VectorXd s(n), xp(n), yp(n);
146 s(0) = 0.;
147
148 for(int i = 0; i < n-1; i++){
149 xp(i) = traj_x(i+1) - traj_x(i);
150 yp(i) = traj_y(i+1) - traj_y(i);
151 acum += sqrt(xp(i)*xp(i) + yp(i)*yp(i));
152 s(i+1) = acum;
153 }
154
155 // Ensure s, xp and yp have the same size by repeating the last X and Y differences
156 xp(n-1) = xp(n-2);
157 yp(n-1) = yp(n-2);
158
159 // Calculate the curvature at each point of the trajectory (k)
160 VectorXd xpp(n), ypp(n), k(n);
161
162 for(int i = 0; i < xp.size()-1; i++){
163 xpp(i) = xp(i+1) - xp(i);
164 ypp(i) = yp(i+1) - yp(i);
165 }
166
167 xpp(n-1) = xpp(n-2);
168 ypp(n-1) = ypp(n-2);
169
170 k = VectorXd::Zero(n);
171
172 for(int i = 0; i < xpp.size() -1; i++){
173 if(xp(i) != yp(i)){
174 k(i) = (xp(i)*ypp(i) - xpp(i)*yp(i)) / pow((xp(i)*xp(i) + yp(i)*yp(i)), 1.5);
175 } else {
176 k(i) = 0.;
177 }
178 }
179
180 MatrixXd res(n,2);
181 res << s, k;
182
183 return res;
184}

187MatrixXd TrajectoryOptimization::generate_speed_and_acc_profile(VectorXd s, VectorXd k){
188 int m = s.size();
189
190 VectorXd speed_profile = VectorXd::Zero(m);
191 speed_profile(0) = kVMax; // Initial max speed for first iteration
192 VectorXd v_grip(m), ds(m);
193 double v_max_braking;
194
195 for(int i = 0; i < m; i++){
196 v_grip(i) = std::min(kVMax, calculate_apex(k(i))); // Max speed to maintain grip
197 }
198
199 // Forwards loop. First iteration. Calculate speed profile and limit by grip
200 for(int i = 1; i < m; i++){
201 ds(i) = s(i) - s(i-1);
202 double ax_max = ggv_ax_throttle(speed_profile(i-1),k(i-1));
203 speed_profile(i) = sqrt(speed_profile(i-1)*speed_profile(i-1) + 2*ax_max*ds(i));
204 if (speed_profile(i) > v_grip(i)){
205 speed_profile(i) = v_grip(i);
206 }
207 }
208 // Close loop
209 speed_profile(0) = speed_profile(m-1);
210
211 // Forwards loop. Second iteration
212 for(int i = 1; i < m; i++){
213 ds(i) = s(i) - s(i-1);
214 double ax_max = ggv_ax_throttle(speed_profile(i-1),k(i-1));
215 speed_profile(i) = sqrt(speed_profile(i-1)*speed_profile(i-1) + 2*ax_max*ds(i));
216 if (speed_profile(i) > v_grip(i)){
217 speed_profile(i) = v_grip(i);
218 }
219 }
220
221 // Backwards loop. First iteration. Limit speed by max braking
222 for(int j = m-1; j > 0; j--){
223 double ax_max_braking = ggv_ax_brake(speed_profile(j),k(j));
224 v_max_braking = sqrt(speed_profile(j)*speed_profile(j) + 2*ax_max_braking*ds(j-1));
225 if (speed_profile(j-1) > v_max_braking){
226 speed_profile(j-1) = v_max_braking;
227 }
228 }
229 // Close loop
230 speed_profile(m-1) = speed_profile(1);
231
232 // Backwards loop. Second iteration
233 for(int j = m-1; j > 0; j--){
234 double ax_max_braking = ggv_ax_brake(speed_profile(j),k(j));
235 v_max_braking = sqrt(speed_profile(j)*speed_profile(j) + 2*ax_max_braking*ds(j-1));
236 if (speed_profile(j-1) > v_max_braking){
237 speed_profile(j-1) = v_max_braking;
238 }
239 }
240
241 // Generate acceleration profile
242 VectorXd acc_profile = VectorXd::Zero(m);
243
244 for(int i = 1; i < m; i++){
245 acc_profile(i) = (speed_profile(i)*speed_profile(i) - speed_profile(i-1)*speed_profile(i-1)) / (2*ds(i-1));
246 }
247 // Close loop
248 acc_profile(1) = acc_profile(m-1);
249
250 MatrixXd res(m,2);
251 res << speed_profile, acc_profile;
252
253 return res;
254}

257double TrajectoryOptimization::calculate_apex(double k){
258
259 if (std::abs(k) < 0.01) {k = 0.01;}
260 double v_grip = std::sqrt(kMuY * kG / (std::abs(k) - kMuY * kCLift / kMass));
261 return v_grip;
262}

265double TrajectoryOptimization::ggv_ax_throttle(double v, double k){
266 double ay_max_v = kMuY * (kG + kCLift/kMass * v*v);
267 double ax_max_v = kMuXThrottle * kG + (kMuXThrottle * kCLift - kCDrag) / kMass *v*v;
268
269 double ay = (std::abs(v*v*k) < ay_max_v) ? std::abs(v*v*k) : ay_max_v;
270 double ax_max = ax_max_v * sqrt(1 - std::pow((ay / ay_max_v),2));
271 return ax_max;
272}

275double TrajectoryOptimization::ggv_ax_brake(double v, double k){
276 double ay_max_v = kMuY * (kG + kCLift/kMass * v*v);
277 double ax_max_v = kMuxBrake * kG + (kMuXThrottle * kCLift + kCDrag) / kMass *v*v;
278
279 double ay = (std::abs(v*v*k) < ay_max_v) ? std::abs(v*v*k) : ay_max_v;
280 double ax_max = ax_max_v - ax_max_v * ay / ay_max_v;
281 return ax_max;
282}
