837template <typename T, typename TNearPointLocator>
838template <
839 typename TVertexIter,
840 typename TGetVertexCoordX,
841 typename TGetVertexCoordY>
842void Triangulation<T, TNearPointLocator>::insertVertices(
843 const TVertexIter first,
844 const TVertexIter last,
845 TGetVertexCoordX getX,
846 TGetVertexCoordY getY)
847{
848 if(isFinalized())
849 throw FinalizedError(CDT_SOURCE_LOCATION);
850
851 const bool isFirstTime = vertices.empty();
852 const T max = std::numeric_limits<T>::max();
853 Box2d<T> box = {{max, max}, {-max, -max}};
854 if(vertices.empty()) // called first time
855 {
856 box = envelopBox<T>(first, last, getX, getY);
857 addSuperTriangle(box);
858 }
859 tryInitNearestPointLocator();
860
861 const VertInd nExistingVerts = static_cast<VertInd>(vertices.size());
862 const VertInd nVerts =
863 static_cast<VertInd>(nExistingVerts + std::distance(first, last));
864 // optimization, try to pre-allocate tris
865 triangles.reserve(triangles.size() + 2 * nVerts);
866 vertices.reserve(nVerts);
867 m_vertTris.reserve(nVerts);
868 for(TVertexIter it = first; it != last; ++it)
869 addNewVertex(V2d<T>::make(getX(*it), getY(*it)), noNeighbor);
870
871 switch(m_vertexInsertionOrder)
872 {
873 case VertexInsertionOrder::AsProvided:
874 insertVertices_AsProvided(nExistingVerts);
875 break;
876 case VertexInsertionOrder::Auto:
877 isFirstTime ? insertVertices_KDTreeBFS(nExistingVerts, box.min, box.max)
878 : insertVertices_Randomized(nExistingVerts);
879 break;
880 }
881}
