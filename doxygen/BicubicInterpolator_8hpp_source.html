89void BicubicInterpolator<Real>::setupInterpolator() {
90 // Interpolation will be done by multiplying the coordinates by
91 // coefficients.
92
93 a = Matrix44Array(X->size() - 1, Y->size() - 1);
94
95 // We are going to precompute the interpolation coefficients so
96 // that we can interpolate quickly. This requires a 4x4 matrix for each
97 // "patch".
98 Matrix44 Left, Right;
99
100 Left << 1, 0, 0, 0, 0, 0, 1, 0, -3, 3, -2, -1, 2, -2, 1, 1;
101
102 Right << 1, 0, -3, 2, 0, 0, 3, -2, 0, 1, -2, 1, 0, 0, -1, 1;
103
104 for (int i = 0; i < X->size() - 1; i++) {
105 for (int j = 0; j < Y->size() - 1; j++) {
106 Matrix44 F;
107
108 Real f00, f01, f10, f11;
109 Real fx00, fx01, fx10, fx11;
110 Real fy00, fy01, fy10, fy11;
111 Real fxy00, fxy01, fxy10, fxy11;
112
113 Real fm, fp;
114 int im, ip, jm, jp;
115
116 int iN = X->size();
117 int jN = Y->size();
118
119 // function values
120 f00 = (*Z)(i, j); // <<<<<<
121 f01 = (*Z)(i, j + 1); // <<<<<<
122 f10 = (*Z)(i + 1, j); // <<<<<<
123 f11 = (*Z)(i + 1, j + 1); // <<<<<<
124
125 // need to calculate function values and derivatives
126 // at each corner.
127 //
128 // note: interpolation algorithm is derived for the unit square.
129 // so we need to take the derivatives assuming X(i+1) - X(i) =
130 // Y(j+1) - Y(j) = 1
131 //
132 // using finite-difference approximation. central-difference on the
133 // interior, forwared and backward difference on the ends.
134
135 Real xL =
136 (*X)(i + 1) - (*X)(i); // actual side length in x direction
137 Real yL =
138 (*Y)(j + 1) - (*Y)(j); // actual side length in y direction
139 Real dx, dy;
140
141 // x derivatives
142 //
143 // (i,j+1) (i+1,j+1)
144 //
145 // + +
146 //
147 //
148 // + +
149 //
150 // (i,j) (i+1,j)
151
152 im = std::max(i - 1, 0); // get index to the "left" of current i
153 ip = std::min(i + 1,
154 iN - 1); // get index to the "right" of current i
155
156 dx = ((*X)(ip) - (*X)(im)) /
157 xL; // relative distance between "right" and "left" neighbors
158
159 fp = (*Z)(ip, j);
160 fm = (*Z)(im, j);
161 fx00 = (fp - fm) / dx; // <<<<<<
162
163 fp = (*Z)(ip, j + 1);
164 fm = (*Z)(im, j + 1);
165 fx01 = (fp - fm) / dx; // <<<<<<
166
167 im = std::max(i, 0);
168 ip = std::min(i + 2, iN - 1);
169
170 dx = ((*X)(ip) - (*X)(im)) / xL;
171
172 fp = (*Z)(ip, j);
173 fm = (*Z)(im, j);
174 fx10 = (fp - fm) / dx; // <<<<<<
175
176 fp = (*Z)(ip, j + 1);
177 fm = (*Z)(im, j + 1);
178 fx11 = (fp - fm) / dx; // <<<<<<
179
180 // y derivatives
181
182 jm = std::max(j - 1, 0);
183 jp = std::min(j + 1, jN - 1);
184
185 dy = ((*Y)(jp) - (*Y)(jm)) / yL;
186
187 fp = (*Z)(i, jp);
188 fm = (*Z)(i, jm);
189 fy00 = (fp - fm) / dy; // <<<<<<
190
191 fp = (*Z)(i + 1, jp);
192 fm = (*Z)(i + 1, jm);
193 fy10 = (fp - fm) / dy; // <<<<<<
194
195 jm = std::max(j, 0);
196 jp = std::min(j + 2, jN - 1);
197
198 dy = ((*Y)(jp) - (*Y)(jm)) / yL;
199
200 fp = (*Z)(i, jp);
201 fm = (*Z)(i, jm);
202 fy01 = (fp - fm) / dx; // <<<<<<
203
204 fp = (*Z)(i + 1, jp);
205 fm = (*Z)(i + 1, jm);
206 fy11 = (fp - fm) / dx; // <<<<<<
207
208 // xy derivatives
209
210 im = std::max(i - 1, 0);
211 ip = std::min(i + 1, iN - 1);
212 jm = std::max(j - 1, 0);
213 jp = std::min(j + 1, jN - 1);
214
215 dx = ((*X)(ip) - (*X)(im)) / xL;

217 dy = ((*Y)(jp) - (*Y)(jm)) / yL;
218

220 fm = ((*Z)(ip, jm) - (*Z)(im, jm)) / dx;
221 fxy00 = (fp - fm) / dy; // <<<<<<
222
223 jm = std::max(j, 0);
224 jp = std::min(j + 2, jN - 1);
225

229 fm = ((*Z)(ip, jm) - (*Z)(im, jm)) / dx;
230 fxy01 = (fp - fm) / dy; // <<<<<<
231
232 im = std::max(i, 0);
233 ip = std::min(i + 2, iN - 1);
234 jm = std::max(j - 1, 0);
235 jp = std::min(j + 1, jN - 1);
236
237 dx = ((*X)(ip) - (*X)(im)) / xL;
238
239 dy = ((*Y)(jp) - (*Y)(jm)) / yL;
240
241 fp = ((*Z)(ip, jp) - (*Z)(im, jp)) / dx;
242 fm = ((*Z)(ip, jm) - (*Z)(im, jm)) / dx;
243 fxy10 = (fp - fm) / dy; // <<<<<<
244
245 jm = std::max(j, 0);
246 jp = std::min(j + 2, jN - 1);
247

249
250 fp = ((*Z)(ip, jp) - (*Z)(im, jp)) / dx;
251 fm = ((*Z)(ip, jm) - (*Z)(im, jm)) / dx;
252 fxy11 = (fp - fm) / dy; // <<<<<<
253
254 F << f00, f01, fy00, fy01, f10, f11, fy10, fy11, fx00, fx01, fxy00,
255 fxy01, fx10, fx11, fxy10, fxy11;
256
257 a(i, j) = Left * F * Right;
258 }

262template <class Real>
263Real BicubicInterpolator<Real>::operator()(Real x, Real y) const {
264 BASE::checkData();
265
266 // no extrapolation...
267 if (x < this->xView->minCoeff() || x > this->xView->maxCoeff() ||
268 y < this->yView->minCoeff() || y > this->yView->maxCoeff()) {
269 return 0;
270 }
271
272 int i = this->get_x_index_to_left_of(x);
273 int j = this->get_y_index_below(y);
274
275 if (i < 0) i = 0;
276 if (j < 0) j = 0;
277
278 Real xL = (*X)(i + 1) - (*X)(i);
279 Real yL = (*Y)(j + 1) - (*Y)(j);
280
281 // now, create the coordinate vectors (see Wikipedia
282 // https://en.wikipedia.org/wiki/Bicubic_interpolation)
283 RowVector4 vx;
284 vx[0] = 1; // x^0
285 vx[1] = (x - (*X)(i)) / xL; // x^1
286 vx[2] = vx[1] * vx[1]; // x^2
287 vx[3] = vx[2] * vx[1]; // x^3
288
289 ColVector4 vy;
290 vy[0] = 1; // y^0
291 vy[1] = (y - (*Y)(j)) / yL; // y^1
292 vy[2] = vy[1] * vy[1]; // y^2
293 vy[3] = vy[2] * vy[1]; // y^3
294
295 // interpolation is just x*a*y
296
297 return vx * a(i, j) * vy;
298}
