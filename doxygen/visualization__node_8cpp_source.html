79void Visualization::triangulation_callback(const common_msgs::msg::Triangulation::SharedPtr msg)
80{
81 visualization_msgs::msg::MarkerArray marker_array;
82 visualization_msgs::msg::Marker delete_marker;
83 delete_marker.action = visualization_msgs::msg::Marker::DELETEALL;
84 marker_array.markers.push_back(delete_marker);
85 for (int i = 0; i < msg -> simplices.size(); i++){
86 visualization_msgs::msg::Marker marker;
87 marker.header.frame_id = "arussim/world";
88 marker.ns = "triangulation";
89 marker.id = i;
90 marker.type = visualization_msgs::msg::Marker::LINE_STRIP;
91 marker.action = visualization_msgs::msg::Marker::ADD;
92 marker.pose.position.x = 0;
93 marker.pose.position.y = 0;
94 marker.pose.position.z = 0;
95 marker.scale.x = 0.1;
96 marker.scale.y = 0.1;
97 marker.scale.z = 0.1;
98 marker.color.a = kAlpha;
99 marker.color.r = 0.0;
100 marker.color.g = 1.0;
101 marker.color.b = 0.0;
102 marker.lifetime = rclcpp::Duration::from_seconds(0.0);
103 for (int j=0; j<3; j++){
104 geometry_msgs::msg::Point p;
105 p.x = msg->simplices[i].points[j].x;
106 p.y = msg->simplices[i].points[j].y;
107 p.z = 0;
108 marker.points.push_back(p);
109 }
110 geometry_msgs::msg::Point p;
111 p.x = msg->simplices[i].points[0].x;
112 p.y = msg->simplices[i].points[0].y;
113 p.z = 0;
114 marker.points.push_back(p);
115 marker_array.markers.push_back(marker);
116 }
117 triangulation_visualization_pub_->publish(marker_array);
118
119}
