234 //@brief: class to exactly represent the result of a sequence of arithmetic operations as an sequence of values that sum to the result
235 template<typename T, size_t N>
236 class Expansion : private ExpansionBase<T>, public stdx::array<T, N> {
237 private:
238 public:
239 size_t m_size;
240 template <typename S> friend class ExpansionBase;//access for base class
241 template <typename S, size_t M> friend class Expansion;//access for expansions of different size
242
243 Expansion() : m_size(0) {}
244 template <size_t M> Expansion& operator=(const Expansion<T, M>& e) {
245 PREDICATES_PORTABLE_STATIC_ASSERT(M <= N, cannot_assign_a_larger_expansion_to_a_smaller_expansion);
246 stdx::copy_n(e.cbegin(), e.size(), stdx::array<T, N>::begin());
247 m_size = e.size();
248 return *this;
249 }
250
251 //vector like convenience functions
252 size_t size() const {return m_size;}
253 bool empty() const {return 0 == m_size;}
254 void push_back(const T v) {stdx::array<T, N>::operator[](m_size++) = v;}
255
256 public:
257 //estimates of expansion value
258 T mostSignificant() const {return empty() ? T(0) : stdx::array<T, N>::operator[](m_size - 1);}
259 T estimate() const {return std::accumulate(stdx::array<T, N>::cbegin(), stdx::array<T, N>::cbegin() + size(), T(0));}

261 template <size_t M> Expansion<T, N+M> operator+(const Expansion<T, M>& f) const {
262 Expansion<T, N+M> h;
263 h.m_size = ExpansionBase<T>::ExpansionSum(this->data(), this->size(), f.data(), f.size(), h.data());
264 return h;
265 }

268 Expansion operator-() const {Expansion e = *this; e.negate(); return e;}
269 template <size_t M> Expansion<T, N+M> operator-(const Expansion<T, M>& f) const {return operator+(-f);}

285 //@param a: lhs item to compare
286 //@param b: rhs item to compare

292 class ExpansionBase {
293 private:
294 static const T Splitter;

299 //combine result + roundoff error into expansion
300 static inline Expansion<T, 2> MakeExpansion(const T value, const T tail) {
301 Expansion<T, 2> e;
302 if(T(0) != tail) e.push_back(tail);
303 if(T(0) != value) e.push_back(value);
304 return e;
305 }

307 protected:
308 //add 2 expansions
309 static size_t ExpansionSum(T const * const e, const size_t n, T const * const f, const size_t m, T * const h) {
310 std::merge(e, e + n, f, f + m, h, absLess<T>);
311 if(m == 0) return n;
312 if(n == 0) return m;
313 size_t hIndex = 0;
314 T Q = h[0];
315 T Qnew = h[1] + Q;
316 T hh = FastPlusTail(h[1], Q, Qnew);
317 Q = Qnew;
318 if(T(0) != hh) h[hIndex++] = hh;
319 for(size_t g = 2; g != n + m; ++g) {
320 Qnew = Q + h[g];
321 hh = PlusTail(Q, h[g], Qnew);
322 Q = Qnew;
323 if(T(0) != hh) h[hIndex++] = hh;
324 }
325 if(T(0) != Q) h[hIndex++] = Q;
326 return hIndex;
327 }
328
329 //scale an expansion by a constant
330 static size_t ScaleExpansion(T const * const e, const size_t n, const T b, T * const h) {
331 if(n == 0 || T(0) == b) return 0;
332 size_t hIndex = 0;
333 T Q = e[0] * b;
334 const std::pair<T, T> bSplit = Split(b);
335 T hh = MultTailPreSplit(e[0], b, bSplit, Q);
336 if(T(0) != hh) h[hIndex++] = hh;
337 for(size_t eIndex = 1; eIndex < n; ++eIndex) {
338 T Ti = e[eIndex] * b;
339 T ti = MultTailPreSplit(e[eIndex], b, bSplit, Ti);
340 T Qi = Q + ti;
341 hh = PlusTail(Q, ti, Qi);
342 if(T(0) != hh) h[hIndex++] = hh;
343 Q = Ti + Qi;
344 hh = FastPlusTail(Ti, Qi, Q);
345 if(T(0) != hh) h[hIndex++] = hh;
346 }
347 if(T(0) != Q) h[hIndex++] = Q;
348 return hIndex;
349 }

351 public:
352 //roundoff error of x = a + b
353 static inline T PlusTail(const T a, const T b, const T x) {
354 const T bVirtual = x - a;
355 const T aVirtual = x - bVirtual;
356 const T bRoundoff = b - bVirtual;
357 const T aRoundoff = a - aVirtual;
358 return aRoundoff + bRoundoff;
359 }
360
361 //roundoff error of x = a + b if |a| > |b|
362 static inline T FastPlusTail(const T a, const T b, const T x) {
363 const T bVirtual = x - a;
364 return b - bVirtual;
365 }
366
367 //roundoff error of x = a - b
368 static inline T MinusTail(const T a, const T b, const T x) {
369 const T bVirtual = a - x;
370 const T aVirtual = x + bVirtual;
371 const T bRoundoff = bVirtual - b;
372 const T aRoundoff = a - aVirtual;
373 return aRoundoff + bRoundoff;
374 }

376 //split a into 2 nonoverlapping values
377 static inline std::pair<T, T> Split(const T a) {
378 const T c = a * Splitter;
379 const T aBig = c - a;
380 const T aHi = c - aBig;
381 return std::pair<T, T>(aHi, a - aHi);
382 }

384 //roundoff error of x = a * b via dekkers product
385 static inline T DekkersProduct(const T /*a*/, const std::pair<T, T> aSplit, const T /*b*/, const std::pair<T, T> bSplit, const T p) {
386 T y = p - T(aSplit.first * bSplit.first);
387 y -= T(aSplit.second * bSplit.first);
388 y -= T(aSplit.first * bSplit.second);
389 return T(aSplit.second * bSplit.second) - y;
390 }

393#if defined(PREDICATES_CXX11_IS_SUPPORTED) && defined(PREDICATES_FAST_FMA)
394 static T MultTail(const T a, const T b, const T p) {return std::fma(a, b, -p);}

396#else
397 static T MultTail(const T a, const T b, const T p) {return DekkersProduct(a, Split(a), b, Split(b), p);}
398 static T MultTailPreSplit(const T a, const T b, const std::pair<T, T> bSplit, const T p) {return DekkersProduct(a, Split(a), b, bSplit, p);}

415 //expand the determinant of {{ax, ay}, {bx, by}} (unrolled Mult(ax, by) - Mult(ay, bx))
416 static inline Expansion<T, 4> TwoTwoDiff(const T ax, const T by, const T ay, const T bx) {
417 const T axby1 = ax * by;
418 const T axby0 = MultTail(ax, by, axby1);
419 const T bxay1 = bx * ay;
420 const T bxay0 = MultTail(bx, ay, bxay1);
421 const T _i0 = axby0 - bxay0;
422 const T x0 = MinusTail(axby0, bxay0, _i0);
423 const T _j = axby1 + _i0;
424 const T _0 = PlusTail(axby1, _i0, _j);
425 const T _i1 = _0 - bxay1;
426 const T x1 = MinusTail(_0, bxay1, _i1);
427 const T x3 = _j + _i1;
428 const T x2 = PlusTail(_j, _i1, x3);
429 Expansion<T, 4> e;
430 if(T(0) != x0) e.push_back(x0);
431 if(T(0) != x1) e.push_back(x1);
432 if(T(0) != x2) e.push_back(x2);
433 if(T(0) != x3) e.push_back(x3);
434 return e;
435 }

437 //TwoTwoDiff checking for zeros to avoid extra splitting
438 static inline Expansion<T, 4> TwoTwoDiffZeroCheck(const T ax, const T by, const T ay, const T bx) {
439 Expansion<T, 4> e;
440 if(T(0) == ax && T(0) == ay) return e;
441 else if(T(0) == ax) e = Mult(ay, bx);
442 else if(T(0) == ay) e = Mult(ax, by);
443 else e = TwoTwoDiff(ax, by, ay, bx);
444 return e;
445 }

528 //@brief : determine if the 3d point e is inside, on, or outside the sphere defined by a, b, c, and d
529 //@param pa: pointer to a as {x, y, z}
530 //@param pb: pointer to b as {x, y, z}

532 //@param pd: pointer to d as {x, y, z}
533 //@param pe: pointer to e as {x, y, z}
534 //@return : determinant of {{ax - ex, ay - ey, az - ez, (ax - ex)^2 + (ay - ey)^2 + (az - ez)^2}, {bx - ex, by - ey, bz - ez, (bx - ex)^2 + (by - ey)^2 + (bz - ez)^2}, {cx - ex, cy - ey, cz - ez, (cx - ex)^2 + (cy - ey)^2 + (cz - ez)^2}, {dx - ex, dy - ey, dz - ez, (dx - ex)^2 + (dy - ey)^2 + (dz - ez)^2}}

536 template <typename T> T insphere(T const*const pa, T const*const pb, T const*const pc, T const*const pd, T const*const pe) {
537 const detail::Expansion<T, 4> ab = detail::ExpansionBase<T>::TwoTwoDiff(pa[0], pb[1], pb[0], pa[1]);
538 const detail::Expansion<T, 4> bc = detail::ExpansionBase<T>::TwoTwoDiff(pb[0], pc[1], pc[0], pb[1]);
539 const detail::Expansion<T, 4> cd = detail::ExpansionBase<T>::TwoTwoDiff(pc[0], pd[1], pd[0], pc[1]);
540 const detail::Expansion<T, 4> de = detail::ExpansionBase<T>::TwoTwoDiff(pd[0], pe[1], pe[0], pd[1]);
541 const detail::Expansion<T, 4> ea = detail::ExpansionBase<T>::TwoTwoDiff(pe[0], pa[1], pa[0], pe[1]);
542 const detail::Expansion<T, 4> ac = detail::ExpansionBase<T>::TwoTwoDiff(pa[0], pc[1], pc[0], pa[1]);
543 const detail::Expansion<T, 4> bd = detail::ExpansionBase<T>::TwoTwoDiff(pb[0], pd[1], pd[0], pb[1]);
544 const detail::Expansion<T, 4> ce = detail::ExpansionBase<T>::TwoTwoDiff(pc[0], pe[1], pe[0], pc[1]);
545 const detail::Expansion<T, 4> da = detail::ExpansionBase<T>::TwoTwoDiff(pd[0], pa[1], pa[0], pd[1]);
546 const detail::Expansion<T, 4> eb = detail::ExpansionBase<T>::TwoTwoDiff(pe[0], pb[1], pb[0], pe[1]);
547
548 const detail::Expansion<T, 24> abc = bc * pa[2] + ac * -pb[2] + ab * pc[2];
549 const detail::Expansion<T, 24> bcd = cd * pb[2] + bd * -pc[2] + bc * pd[2];
550 const detail::Expansion<T, 24> cde = de * pc[2] + ce * -pd[2] + cd * pe[2];
551 const detail::Expansion<T, 24> dea = ea * pd[2] + da * -pe[2] + de * pa[2];
552 const detail::Expansion<T, 24> eab = ab * pe[2] + eb * -pa[2] + ea * pb[2];
553 const detail::Expansion<T, 24> abd = bd * pa[2] + da * pb[2] + ab * pd[2];
554 const detail::Expansion<T, 24> bce = ce * pb[2] + eb * pc[2] + bc * pe[2];
555 const detail::Expansion<T, 24> cda = da * pc[2] + ac * pd[2] + cd * pa[2];
556 const detail::Expansion<T, 24> deb = eb * pd[2] + bd * pe[2] + de * pb[2];
557 const detail::Expansion<T, 24> eac = ac * pe[2] + ce * pa[2] + ea * pc[2];
558
559 const detail::Expansion<T, 96> bcde = (cde + bce) - (deb + bcd);
560 const detail::Expansion<T, 96> cdea = (dea + cda) - (eac + cde);
561 const detail::Expansion<T, 96> deab = (eab + deb) - (abd + dea);
562 const detail::Expansion<T, 96> eabc = (abc + eac) - (bce + eab);
563 const detail::Expansion<T, 96> abcd = (bcd + abd) - (cda + abc);
564
565 const detail::Expansion<T, 1152> adet = bcde * pa[0] * pa[0] + bcde * pa[1] * pa[1] + bcde * pa[2] * pa[2];
566 const detail::Expansion<T, 1152> bdet = cdea * pb[0] * pb[0] + cdea * pb[1] * pb[1] + cdea * pb[2] * pb[2];
567 const detail::Expansion<T, 1152> cdet = deab * pc[0] * pc[0] + deab * pc[1] * pc[1] + deab * pc[2] * pc[2];
568 const detail::Expansion<T, 1152> ddet = eabc * pd[0] * pd[0] + eabc * pd[1] * pd[1] + eabc * pd[2] * pd[2];
569 const detail::Expansion<T, 1152> edet = abcd * pe[0] * pe[0] + abcd * pe[1] * pe[1] + abcd * pe[2] * pe[2];
570
571 const detail::Expansion<T, 5760> deter = (adet + bdet) + ((cdet + ddet) + edet);
572 return deter.mostSignificant();
573 }

614 template <typename T> T orient2d(T const ax, T const ay, T const bx, T const by, T const cx, T const cy) {
615 const T acx = ax - cx;
616 const T bcx = bx - cx;
617 const T acy = ay - cy;
618 const T bcy = by - cy;
619 const T detleft = acx * bcy;
620 const T detright = acy * bcx;
621 T det = detleft - detright;
622 if((detleft < 0) != (detright < 0)) return det;
623 if(T(0) == detleft || T(0) == detright) return det;
624
625 const T detsum = std::abs(detleft + detright);
626 T errbound = Constants<T>::ccwerrboundA * detsum;
627 if(std::abs(det) >= std::abs(errbound)) return det;
628
629 const detail::Expansion<T, 4> B = detail::ExpansionBase<T>::TwoTwoDiff(acx, bcy, acy, bcx);
630 det = B.estimate();
631 errbound = Constants<T>::ccwerrboundB * detsum;
632 if(std::abs(det) >= std::abs(errbound)) return det;
633
634 const T acxtail = detail::ExpansionBase<T>::MinusTail(ax, cx, acx);
635 const T bcxtail = detail::ExpansionBase<T>::MinusTail(bx, cx, bcx);
636 const T acytail = detail::ExpansionBase<T>::MinusTail(ay, cy, acy);
637 const T bcytail = detail::ExpansionBase<T>::MinusTail(by, cy, bcy);
638 if(T(0) == acxtail && T(0) == bcxtail && T(0) == acytail && T(0) == bcytail) return det;
639
640 errbound = Constants<T>::ccwerrboundC * detsum + Constants<T>::resulterrbound * std::abs(det);
641 det += (acx * bcytail + bcy * acxtail) - (acy * bcxtail + bcx * acytail);
642 if(std::abs(det) >= std::abs(errbound)) return det;
643
644 const detail::Expansion<T, 16> D = ((B + detail::ExpansionBase<T>::TwoTwoDiff(acxtail, bcy, acytail, bcx)) + detail::ExpansionBase<T>::TwoTwoDiff(acx, bcytail, acy, bcxtail)) + detail::ExpansionBase<T>::TwoTwoDiff(acxtail, bcytail, acytail, bcxtail);
645 return D.mostSignificant();
646 }

652 template <typename T> T incircle(T const ax, T const ay, T const bx, T const by, T const cx, T const cy, T const dx, T const dy) {
653 const T adx = ax - dx;
654 const T bdx = bx - dx;
655 const T cdx = cx - dx;
656 const T ady = ay - dy;
657 const T bdy = by - dy;
658 const T cdy = cy - dy;
659 const T bdxcdy = bdx * cdy;
660 const T cdxbdy = cdx * bdy;
661 const T cdxady = cdx * ady;
662 const T adxcdy = adx * cdy;
663 const T adxbdy = adx * bdy;
664 const T bdxady = bdx * ady;
665 const T alift = adx * adx + ady * ady;
666 const T blift = bdx * bdx + bdy * bdy;
667 const T clift = cdx * cdx + cdy * cdy;
668 T det = alift * (bdxcdy - cdxbdy) + blift * (cdxady - adxcdy) + clift * (adxbdy - bdxady);
669 const T permanent = (std::abs(bdxcdy) + std::abs(cdxbdy)) * alift
670 + (std::abs(cdxady) + std::abs(adxcdy)) * blift
671 + (std::abs(adxbdy) + std::abs(bdxady)) * clift;
672 T errbound = Constants<T>::iccerrboundA * permanent;
673 if(std::abs(det) >= std::abs(errbound)) return det;
674
675 const detail::Expansion<T, 4> bc = detail::ExpansionBase<T>::TwoTwoDiff(bdx, cdy, cdx, bdy);
676 const detail::Expansion<T, 4> ca = detail::ExpansionBase<T>::TwoTwoDiff(cdx, ady, adx, cdy);
677 const detail::Expansion<T, 4> ab = detail::ExpansionBase<T>::TwoTwoDiff(adx, bdy, bdx, ady);
678 const detail::Expansion<T, 32> adet = bc * adx * adx + bc * ady * ady;
679 const detail::Expansion<T, 32> bdet = ca * bdx * bdx + ca * bdy * bdy;
680 const detail::Expansion<T, 32> cdet = ab * cdx * cdx + ab * cdy * cdy;
681 const detail::Expansion<T, 96> fin1 = adet + bdet + cdet;
682 det = fin1.estimate();
683 errbound = Constants<T>::iccerrboundB * permanent;
684 if(std::abs(det) >= std::abs(errbound)) return det;
685
686 const T adxtail = detail::ExpansionBase<T>::MinusTail(ax, dx, adx);
687 const T adytail = detail::ExpansionBase<T>::MinusTail(ay, dy, ady);
688 const T bdxtail = detail::ExpansionBase<T>::MinusTail(bx, dx, bdx);
689 const T bdytail = detail::ExpansionBase<T>::MinusTail(by, dy, bdy);
690 const T cdxtail = detail::ExpansionBase<T>::MinusTail(cx, dx, cdx);
691 const T cdytail = detail::ExpansionBase<T>::MinusTail(cy, dy, cdy);
692 if(T(0) == adxtail && T(0) == bdxtail && T(0) == cdxtail && T(0) == adytail && T(0) == bdytail && T(0) == cdytail) return det;
693
694 errbound = Constants<T>::iccerrboundC * permanent + Constants<T>::resulterrbound * std::abs(det);
695 det += ((adx * adx + ady * ady) * ((bdx * cdytail + cdy * bdxtail) - (bdy * cdxtail + cdx * bdytail))
696 + (bdx * cdy - bdy * cdx) * (adx * adxtail + ady * adytail) * T(2))
697 + ((bdx * bdx + bdy * bdy) * ((cdx * adytail + ady * cdxtail) - (cdy * adxtail + adx * cdytail))
698 + (cdx * ady - cdy * adx) * (bdx * bdxtail + bdy * bdytail) * T(2))
699 + ((cdx * cdx + cdy * cdy) * ((adx * bdytail + bdy * adxtail) - (ady * bdxtail + bdx * adytail))
700 + (adx * bdy - ady * bdx) * (cdx * cdxtail + cdy * cdytail) * T(2));
701 if(std::abs(det) >= std::abs(errbound)) return det;
702 return exact::incircle(ax, ay, bx, by, cx, cy, dx, dy);
703 }

709 //@brief : determine if the 3d point d is above, on, or below the plane defined by a, b, and c
710 //@param pa: pointer to a as {x, y, z}
711 //@param pb: pointer to b as {x, y, z}
712 //@param pc: pointer to c as {x, y, z}
713 //@param pd: pointer to d as {x, y, z}
714 //@return : determinant of {{ax - dx, ay - dy, az - dz}, {bx - dx, by - dy, bz - dz}, {cx - dx, cy - dy, cz - dz}}
715 //@note : positive, 0, negative result for c above, on, or below the plane defined by a, b, and c
716 template <typename T> T orient3d(T const*const pa, T const*const pb, T const*const pc, T const*const pd) {
717 const T adx = pa[0] - pd[0];
718 const T bdx = pb[0] - pd[0];
719 const T cdx = pc[0] - pd[0];
720 const T ady = pa[1] - pd[1];
721 const T bdy = pb[1] - pd[1];
722 const T cdy = pc[1] - pd[1];
723 const T adz = pa[2] - pd[2];
724 const T bdz = pb[2] - pd[2];
725 const T cdz = pc[2] - pd[2];
726 const T bdxcdy = bdx * cdy;
727 const T cdxbdy = cdx * bdy;
728 const T cdxady = cdx * ady;
729 const T adxcdy = adx * cdy;
730 const T adxbdy = adx * bdy;
731 const T bdxady = bdx * ady;
732 T det = adz * (bdxcdy - cdxbdy) + bdz * (cdxady - adxcdy) + cdz * (adxbdy - bdxady);
733 const T permanent = (std::abs(bdxcdy) + std::abs(cdxbdy)) * std::abs(adz) + (std::abs(cdxady) + std::abs(adxcdy)) * std::abs(bdz) + (std::abs(adxbdy) + std::abs(bdxady)) * std::abs(cdz);
734 T errbound = Constants<T>::o3derrboundA * permanent;
735 if(std::abs(det) >= std::abs(errbound)) return det;
736
737 const detail::Expansion<T, 4> bc = detail::ExpansionBase<T>::TwoTwoDiff(bdx, cdy, cdx, bdy);
738 const detail::Expansion<T, 4> ca = detail::ExpansionBase<T>::TwoTwoDiff(cdx, ady, adx, cdy);
739 const detail::Expansion<T, 4> ab = detail::ExpansionBase<T>::TwoTwoDiff(adx, bdy, bdx, ady);
740 const detail::Expansion<T, 24> fin1 = (bc * adz + ca * bdz) + ab * cdz;
741 det = fin1.estimate();
742 errbound = Constants<T>::o3derrboundB * permanent;
743 if(std::abs(det) >= std::abs(errbound)) return det;
744
745 const T adxtail = detail::ExpansionBase<T>::MinusTail(pa[0], pd[0], adx);
746 const T bdxtail = detail::ExpansionBase<T>::MinusTail(pb[0], pd[0], bdx);
747 const T cdxtail = detail::ExpansionBase<T>::MinusTail(pc[0], pd[0], cdx);
748 const T adytail = detail::ExpansionBase<T>::MinusTail(pa[1], pd[1], ady);
749 const T bdytail = detail::ExpansionBase<T>::MinusTail(pb[1], pd[1], bdy);
750 const T cdytail = detail::ExpansionBase<T>::MinusTail(pc[1], pd[1], cdy);
751 const T adztail = detail::ExpansionBase<T>::MinusTail(pa[2], pd[2], adz);
752 const T bdztail = detail::ExpansionBase<T>::MinusTail(pb[2], pd[2], bdz);
753 const T cdztail = detail::ExpansionBase<T>::MinusTail(pc[2], pd[2], cdz);
754 if(T(0) == adxtail && T(0) == adytail && T(0) == adztail &&
755 T(0) == bdxtail && T(0) == bdytail && T(0) == bdztail &&
756 T(0) == cdxtail && T(0) == cdytail && T(0) == cdztail) return det;
757
758 errbound = Constants<T>::o3derrboundC * permanent + Constants<T>::resulterrbound * std::abs(det);
759 det += (adz * ((bdx * cdytail + cdy * bdxtail) - (bdy * cdxtail + cdx * bdytail)) + adztail * (bdx * cdy - bdy * cdx))
760 + (bdz * ((cdx * adytail + ady * cdxtail) - (cdy * adxtail + adx * cdytail)) + bdztail * (cdx * ady - cdy * adx))
761 + (cdz * ((adx * bdytail + bdy * adxtail) - (ady * bdxtail + bdx * adytail)) + cdztail * (adx * bdy - ady * bdx));
762 if(std::abs(det) >= std::abs(errbound)) return det;
763
764 const detail::Expansion<T, 8> bct = detail::ExpansionBase<T>::TwoTwoDiffZeroCheck(bdxtail, cdy, bdytail, cdx) + detail::ExpansionBase<T>::TwoTwoDiffZeroCheck(cdytail, bdx, cdxtail, bdy);
765 const detail::Expansion<T, 8> cat = detail::ExpansionBase<T>::TwoTwoDiffZeroCheck(cdxtail, ady, cdytail, adx) + detail::ExpansionBase<T>::TwoTwoDiffZeroCheck(adytail, cdx, adxtail, cdy);
766 const detail::Expansion<T, 8> abt = detail::ExpansionBase<T>::TwoTwoDiffZeroCheck(adxtail, bdy, adytail, bdx) + detail::ExpansionBase<T>::TwoTwoDiffZeroCheck(bdytail, adx, bdxtail, ady);
767 const detail::Expansion<T, 192> fin2 = fin1 + bct * adz + cat * bdz + abt * cdz + bc * adztail + ca * bdztail + ab * cdztail
