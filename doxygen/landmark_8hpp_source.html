107 void update_color(int min_color_observations, double min_prob) {
108 if (num_color_observations_ == 0 || world_position_.norm() < 7.0) {
109 color_ = UNCOLORED;
110 return;
111 }
112
113 double avg_blue = sum_prob_blue_ / num_color_observations_;
114 double avg_yellow = sum_prob_yellow_ / num_color_observations_;
115 if (num_color_observations_ >= min_color_observations) {
116 if (avg_blue > min_prob && avg_blue > avg_yellow) {
117 color_ = BLUE;
118 } else if (avg_yellow > min_prob && avg_yellow > avg_blue) {
119 color_ = YELLOW;
120 } else {
121 color_ = UNCOLORED;
122 }
123 }
