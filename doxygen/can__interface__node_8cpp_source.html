236{

250 // Convert frame ID to string (hex format)

263

284 config.endByte = std::stoi(main_vector[1]);

301void CanInterface::parse_msg(const struct can_frame& frame, const CANParseConfig& config) {
302 int32_t rawValue = 0;
303
304 // Calculate the number of bytes
305 uint8_t numBytes = config.endByte - config.startByte + 1;
306
307 if (numBytes == 1) {
308 if (config.isSigned == "yes") {
309 rawValue = static_cast<int8_t>(frame.data[config.startByte]);
310 } else {
311 rawValue = static_cast<uint8_t>(frame.data[config.startByte]);
312 }
313 }
314 else if (numBytes == 2) {
315 if (config.isSigned == "yes") {
316 rawValue = static_cast<int16_t>(
317 (frame.data[config.endByte] << 8) | frame.data[config.startByte]);
318 } else {
319 rawValue = static_cast<uint16_t>(
320 (frame.data[config.endByte] << 8) | frame.data[config.startByte]);
321 }
322
323 }
324 else if (numBytes == 3) {
325 if (config.isSigned == "yes") {
326 rawValue = static_cast<int32_t>(
327 (frame.data[config.startByte + 2] << 16) |
328 (frame.data[config.startByte + 1] << 8) |
329 frame.data[config.startByte]);
330
331 // Sign-extend 24-bit value to 32-bit
332 if (rawValue & 0x00800000) { // Check if the sign bit (23rd bit) is set
333 rawValue |= 0xFF000000; // Extend the sign to the upper bits
334 }
335 } else {
336 rawValue = static_cast<uint32_t>(
337 (frame.data[config.startByte + 2] << 16) |
338 (frame.data[config.startByte + 1] << 8) |
339 frame.data[config.startByte]);
340 }
341 }
342 else if (numBytes == 4) {
343 if (config.isSigned == "yes") {
344 rawValue = static_cast<int32_t>(
345 (frame.data[config.startByte + 3] << 24) |
346 (frame.data[config.startByte + 2] << 16) |
347 (frame.data[config.startByte + 1] << 8) |
348 frame.data[config.startByte]);
349 } else {
350 rawValue = static_cast<uint32_t>(
351 (frame.data[config.startByte + 3] << 24) |
352 (frame.data[config.startByte + 2] << 16) |
353 (frame.data[config.startByte + 1] << 8) |
354 frame.data[config.startByte]);
355 }
356 }
357 else if (kDebug) {
358 RCLCPP_ERROR(
359 this->get_logger(),
360 "Unsupported byte range: %d bytes",
361 numBytes
362 );
363 }
364
365 float scaledValue = std::pow(static_cast<float>(rawValue), config.power) * config.scale + config.offset;
366
367 // Find the associated publisher, create the message and publish the scaled value.
368 auto pub_iter = publishers_.find(config.key);
369 if (pub_iter != publishers_.end()) {
370 std_msgs::msg::Float32 msg;
371 msg.data = scaledValue;
372 pub_iter->second->publish(msg);
373 } else if (kDebug) {
374 RCLCPP_ERROR(this->get_logger(),
375 "No matching publisher in publishers_ for key: %s",
376 config.key.c_str());
377 }

578void CanInterface::send_dl502()
579{
580 struct can_frame frame;
581 frame.can_id = 0x502;
582 frame.can_dlc = 5;
583
584 uint64_t packed_message = 0;
585
586 // Pack each field into the correct bit positions
587 packed_message |= (as_status_ & 0x07) << 0; // Bits 0-2: AS_status (3 bits)
588 packed_message |= (asb_ebs_state_ & 0x07) << 3; // Bits 3-4: ASB_EBS_state (3 bits)
589 packed_message |= (ami_state_ & 0x07) << 5; // Bits 5-7: AMI_state (3 bits)
590 packed_message |= (steering_state_ & 0x01) << 8; // Bit 8: Steering state (1 bit)
591 packed_message |= (asb_redundancy_state_ & 0x07) << 9; // Bits 9-10: ASB_Redundancy_state (3 bits)
592 packed_message |= (lap_counter_ & 0x0F) << 11; // Bits 11-14: Lap counter (4 bits)
593 packed_message |= (cones_count_actual_ & 0xFF) << 15; // Bits 15-22: Cones_count_actual (8 bits)
594 packed_message |= (cones_count_all_ & 0xFFFF) << 23; // Bits 23-39: Cones_count_all (16 bits)
595
596 // Extract the packed message into the frame data bytes (Little-Endian)
597 frame.data[0] = packed_message & 0xFF;
598 frame.data[1] = (packed_message >> 8) & 0xFF;
599 frame.data[2] = (packed_message >> 16) & 0xFF;
600 frame.data[3] = (packed_message >> 24) & 0xFF;
601 frame.data[4] = (packed_message >> 32) & 0xFF;
602
603 write(socket_can1_, &frame, sizeof(struct can_frame));
604}
