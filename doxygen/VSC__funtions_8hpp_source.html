69 //keyhandle_ = VCS_OpenDevice("EPOS4", "MAXON SERIAL V2", "USB", "USB0", &pErrorCode);

72 if (keyhandle_ == nullptr) {
73 RCLCPP_ERROR(logger_, "Error: Failed to open EPOS4 device.");

170 void move_to(double wheel_angle) {
171 uint32_t pErrorCode = 0;
172
173 // 4096 incs = 2pi rad
174 // 5:1 steering ratio
175 // 66:1 motor reduction
176 int motor_position = static_cast<int>(wheel_angle*(4096 * 5 * 66)/(2*M_PI));
177
178 if (_is_enable_) {
179
180 int ret = VCS_MoveToPosition(keyhandle_, NodeID, motor_position, 1, 1, &pErrorCode);
181
182 if (ret == 0) {
183 RCLCPP_ERROR(logger_, "Error: MoveToPosition failed with error code %d. Disabling controller", pErrorCode);
184 disable();
185 }
186 } else {
187 RCLCPP_ERROR(logger_, "Error: Cannot move to position with disabled controller.");
188 }
189 }

196 std::vector<double> get_epos_info() {
197 uint32_t pErrorCode = 0;
198 uint32_t pMovementState = 0;
199 int32_t pPosition = 0;
200 int32_t pTargetPosition = 0;
201 int32_t pVelocity = 0;
202 int32_t pVelocityAvg = 0;
203 int16_t pTorque = 0;
204 uint32_t pBytesRead = 0;
205
206 std::vector<double> epos_info;
207
208 if (_is_enable_) {
209 // Get Movement State
210 if (VCS_GetMovementState(keyhandle_, NodeID, &pMovementState, &pErrorCode) == 0) {
211 RCLCPP_WARN(logger_, "WARNING: getMovementState error with code %d. Disabling controller.", pErrorCode);
212 disable();
213 }
214 epos_info.push_back(static_cast<double>(pMovementState));
215
216 // Get Position
217 if (VCS_GetPositionIs(keyhandle_, NodeID, &pPosition, &pErrorCode) == 0) {
218 RCLCPP_WARN(logger_, "WARNING: getPosition error with code %d. Disabling controller.", pErrorCode);
219 disable();
220 }
221 if (std::bitset<32>(pPosition).count() == 34) {
222 pPosition -= (1LL << 32); // Adjust for overflow
223 }
224 epos_info.push_back((pPosition*2*M_PI)/(4096 * 5 * 66));
225
226 // Get Target Position
227 if (VCS_GetTargetPosition(keyhandle_, NodeID, &pTargetPosition, &pErrorCode) == 0) {
228 RCLCPP_WARN(logger_, "WARNING: getTargetPosition error with code %d. Disabling controller.", pErrorCode);
229 disable();
230 }
231 if (std::bitset<32>(pTargetPosition).count() == 34) {
232 pTargetPosition -= (1LL << 32);
233 }
234 epos_info.push_back((pTargetPosition*2*M_PI)/(4096 * 5 * 66));
235
236 // Get Velocity
237 if (VCS_GetVelocityIs(keyhandle_, NodeID, &pVelocity, &pErrorCode) == 0) {
238 RCLCPP_WARN(logger_, "WARNING: getVelocity error with code %d. Disabling controller.", pErrorCode);
239 disable();
240 }
241 epos_info.push_back(static_cast<double>(pVelocity));
242
243 // Get Averaged Velocity
244 if (VCS_GetVelocityIsAveraged(keyhandle_, NodeID, &pVelocityAvg, &pErrorCode) == 0) {
245 RCLCPP_WARN(logger_, "WARNING: getVelocityAvg error with code %d. Disabling controller.", pErrorCode);
246 disable();
247 }
248 epos_info.push_back(static_cast<double>(pVelocityAvg));
249
250 // Get Torque
251 if (VCS_GetObject(keyhandle_, NodeID, 0x6077, 0x00, &pTorque, 2, &pBytesRead, &pErrorCode) == 0) {
252 RCLCPP_WARN(logger_, "WARNING: getTorque error with code %d. Disabling controller.", pErrorCode);
253 disable();
254 }
255 epos_info.push_back(static_cast<double>(pTorque));
256
257 return epos_info;
258 } else {
259 RCLCPP_ERROR(logger_, "ERROR: Cannot get epos info with disabled controller :(");
260 return {};
261 }
262 }
