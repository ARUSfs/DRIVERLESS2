37 // Define cost matrices
38
39 Q.resize(C.rows(),C.rows());

41 R = kCostSteeringDelta * Eigen::MatrixXd::Identity(kPredictionHorizon, kPredictionHorizon);
42
43 // MPC matrices construction
44
45 Eigen::MatrixXd PSI = Eigen::MatrixXd::Zero(C.rows() * kPredictionHorizon, Ad.cols());
46 Eigen::MatrixXd YPS = Eigen::MatrixXd::Zero(C.rows() * kPredictionHorizon, Bd.cols());
47 Eigen::MatrixXd THETA = Eigen::MatrixXd::Zero(C.rows() * kPredictionHorizon, Bd.cols() * kPredictionHorizon);
48 Eigen::MatrixXd Q_THETA = Eigen::MatrixXd::Zero(Q.rows() * kPredictionHorizon, Bd.cols() * kPredictionHorizon);
49 Eigen::MatrixXd Q_YPS = Eigen::MatrixXd::Zero(Q.rows() * kPredictionHorizon, Bd.cols());
50
51 Eigen::MatrixXd Apow = Eigen::MatrixXd::Identity(Ad.rows(),Ad.cols());
52
53 YPS.topRows(C.rows()) = C * Bd;
54
55 for (int i = 0; i < kPredictionHorizon; i++){
56 if (i>0){YPS.middleRows(i * C.rows(), C.rows()) = YPS.middleRows((i-1)*C.rows(), C.rows()) + C * Apow * Bd;}
57
58 Apow *= Ad;
59
60 linearize_model(prediction_speed_(i+1), A, B);
61
62 discretize_model(A, B, kTsMPC, Ad, Bd);
63
64 PSI.middleRows(i * C.rows(), C.rows()) = C * Apow;
65 Q_YPS.middleRows(i * Q.rows(), Q.rows()) = Q * YPS.middleRows(i * C.rows(), C.rows());
66
67 }
68
69 for (int i = 0; i < kPredictionHorizon; i++){
70 THETA.block(i * C.rows(), i * Bd.cols(), (kPredictionHorizon - i) * C.rows(), Bd.cols()) = YPS.topRows((kPredictionHorizon - i) * C.rows());
71 Q_THETA.block(i * Q.rows(), i * Bd.cols(), (kPredictionHorizon - i) * Q.rows(), Bd.cols()) = Q_YPS.topRows((kPredictionHorizon - i) * Q.rows());
72 }
73
74 // Solve least squares problem
75
76 Eigen::MatrixXd A_mpc = 2 * (Q_THETA.transpose() * THETA + R);
77 Eigen::MatrixXd b_mpc = 2 * Q_THETA.transpose() * (target_trajectory_ - PSI * x_0_ - YPS * delta_);
78
79 U = A_mpc.llt().solve(b_mpc);
80
81 double delta_target = delta_ + U(0,0);
82
83 if (prediction_speed_(0) > 2.0){
84 linearize_model(prediction_speed_(0), A, B);
85 discretize_model(A, B, kTsMPC, Ad, Bd);
86 Eigen::VectorXd x_next = Ad* x_0_ + Bd * delta_target;
87 target_r_ = x_next(3); // Desired yawrate from mpc
88 }
89
90 // Delay compensation
91 for (int i = 1; i < kCompensationSteps + 1; i++){
92 delta_target += U(i,0);
93 }
94
95 return delta_target;

242 void discretize_model(const Eigen::MatrixXd& A, const Eigen::MatrixXd& B, const double& dT, Eigen::MatrixXd& Phi, Eigen::MatrixXd& Gamma) {
243
244 int m = A.rows();
245 int n = A.cols();
246 int nb = B.cols();
247
248 Eigen::MatrixXd tmp3(m + nb, n + nb);

263 Point interpolate_data(const std::vector<Point> &XYdata, const std::vector<float> &vx_profile, const std::vector<float> &s, const double &s0) {
264
265 if (s0 <= s.front()) return XYdata.front();
266 if (s0 >= s.back()) return XYdata.back();
267
268 Point point;
269
270 for (size_t i = 0; i < s.size() - 1; i++) {
271 if (s0 >= s[i] && s0 <= s[i + 1]) {
272 double alpha = (s0 - s[i]) / (s[i + 1] - s[i]);
273 double x_interp = XYdata[i].x + alpha * (XYdata[i + 1].x - XYdata[i].x);
274 double y_interp = XYdata[i].y + alpha * (XYdata[i + 1].y - XYdata[i].y);
275
276 yaw_interp = std::atan2((XYdata[i+1].y-XYdata[i].y),(XYdata[i+1].x-XYdata[i].x));
277
278 v_interp = std::max(2.0, vx_profile[i] + alpha * (vx_profile[i+1] - vx_profile[i]));
279
280 point.x = x_interp;
281 point.y = y_interp;
282 }
283 }
284 return point;
