100 Vector2d estimate_speed(double ax, double r, double delta, double delta_der, double wheelspeed_mean){
101 // Kinematic bicycle
102 VectorXd u_kin(m_kin_), z_kin(p_), x_kin(n_);
103 MatrixXd P_kin(n_,n_);
104
105 M_kin_(1,0) = lr_ / L_ * delta_der;
106 v_filter_kin_.update_model_matrix(M_kin_);
107
108 u_kin << ax, ax*delta;
109 z_kin << wheelspeed_mean, lr_ / L_ * std::tan(delta) * vx_;
110
111 v_filter_kin_.estimate_state(u_kin, z_kin);
112 x_kin = v_filter_kin_.get_estimated_state();
113 P_kin = v_filter_kin_.get_estimated_covariance();
114
115 // Dynamic bicycle
116 VectorXd u_dyn(m_dyn_), z_dyn(p_), x_dyn(n_);
117 MatrixXd P_dyn(n_,n_);
118
119 M_dyn_(0,1) = r;
120 M_dyn_(1,0) = -r;
121 v_filter_dyn_.update_model_matrix(M_dyn_);
122
123 B_dyn_(0,1) = -std::sin(delta);
124 B_dyn_(1,1) = std::cos(delta);
125 v_filter_dyn_.update_control_matrix(B_dyn_);
126
127 double a_f = std::atan2(vy_ + lf_ * r, vx_) - delta;
128 double a_r = std::atan2(vy_ - lr_ * r, vx_) - delta;
129 double F_fy = D_lat_ * std::sin(C_lat_ * std::atan(B_lat_ * a_f));
130 double F_ry = D_lat_ * std::sin(C_lat_ * std::atan(B_lat_ * a_r));
131 u_dyn << ax, F_fy / m_, F_ry / m_;
132
133 z_dyn << wheelspeed_mean,
134 0.25 * (vx_ * (lf_*cf_ + lr_*cr_)) / (m_*vx_*vx_ + cf_*lf_*lf_ + cr_*lr_*lr_) * delta;
135
136 v_filter_dyn_.estimate_state(u_dyn, z_dyn);
137 x_dyn = v_filter_dyn_.get_estimated_state();
138 P_dyn = v_filter_dyn_.get_estimated_covariance();
139
140 // Estimated state
141 double lambda = std::clamp((vx_ - vx_blend_min_) / (vx_blend_max_ - vx_blend_min_), 0., 1.);
142 VectorXd x_est = (1 - lambda) * x_kin + lambda * x_dyn;
143 vx_ = x_est(0);
144 vy_ = x_est(1);
145
146 // Update estimated state and covariance
147 v_filter_kin_.update_state(x_est);
148 v_filter_dyn_.update_state(x_est);
149
150 MatrixXd P_est = (1 - lambda) * P_kin + lambda * P_dyn;
151 v_filter_kin_.update_covariance(P_est);
152 v_filter_dyn_.update_covariance(P_est);
153
154 return x_est;
155 }
