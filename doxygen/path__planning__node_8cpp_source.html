326CDT::Triangulation<double> PathPlanning::create_triangulation(pcl::PointCloud<ConeXYZColorScore> input_cloud){
327 // Initialize empty triangulation and empty points vector
328 CDT::Triangulation<double> triangulation;
329 std::vector<CDT::V2d<double>> points;
330 // Add the points to the triangulation from the cloud in CDT format
331 for (std::size_t i = 0; i<input_cloud.size();i++){
332 ConeXYZColorScore cone = input_cloud.points[i];
333 CDT::V2d<double> point = CDT::V2d<double>::make(cone.x, cone.y);
334 points.push_back(point);
335 }
336 triangulation.insertVertices(points);
337 // Delete the super triangle to avoid errors in next steps
338 triangulation.eraseSuperTriangle();
339 // Iterate over the triangles and delete the ones that don't meet the conditions
340 CDT::TriangleVec triangles = triangulation.triangles;
341 CDT::Triangulation<double>::V2dVec vertices = triangulation.vertices;
342 CDT::TriIndUSet deleted_tri;
343 for (int i=0; i<triangles.size(); i++) {
344 // Angle and length calculations
345 CDT::VerticesArr3 vert_ind = triangles[i].vertices;
346 ConeXYZColorScore a = input_cloud.points[vert_ind[0]];
347 ConeXYZColorScore b = input_cloud.points[vert_ind[1]];
348 ConeXYZColorScore c = input_cloud.points[vert_ind[2]];
349 CDT::V2d<double> ab, bc, ca;
350 double a_angle, b_angle, c_angle;
351 ab = CDT::V2d<double>::make(b.x-a.x, b.y-a.y);
352 bc = CDT::V2d<double>::make(c.x-b.x, c.y-b.y);
353 ca = CDT::V2d<double>::make(a.x-c.x, a.y-c.y);
354
355 // Delete triangles with long edges and all vertices with the same color
356 if (lap_count_==0 && ((a.x == origin_.x and a.y == origin_.y) or // Check if the origin vertex is
357 (b.x == origin_.x and b.y == origin_.y) or // in the triangle and skip it
358 (c.x == origin_.x and c.y == origin_.y))){ // (origin vertex is the car position)
359 continue;
360 }
361
362 if (a.color != UNCOLORED and a.color == b.color and b.color == c.color){
363 deleted_tri.insert(i);
364 } else if (distance(a,b) > kMaxTriLen or distance(b,c) > kMaxTriLen or distance(c,a) > kMaxTriLen){
365 deleted_tri.insert(i);
366 }
367 }
368 triangulation.removeTriangles(deleted_tri);
369 return triangulation;
370}

372std::vector<ConeXYZColorScore> PathPlanning::get_back_edge(){
373
374 std::vector<ConeXYZColorScore> back_edge = {};
375
376 pcl::PointCloud<ConeXYZColorScore> local_cloud = pcl_cloud_;
377 local_cloud.points.push_back(origin_);
378
379 // Create the triangulation
380 CDT::Triangulation<double> triang = this->create_triangulation(local_cloud);
381
382 // Initializing it from each of the origin triangles
383 int orig_index = get_vertex_index(CDT::V2d<double>::make(x_,y_), triang);
384 if (orig_index == -1){
385 if (kDebug) RCLCPP_WARN(this->get_logger(), "Origin vertex not found");
386 return back_edge;
387 }
388 std::vector<int> o_triangles = get_triangles_from_vert(orig_index, triang);
389
390 for (int i = 0; i<o_triangles.size(); i++){
391 // Calculate the centroid of the triangle and the angle difference with the car yaw
392 CDT::V2d<double> centroid = compute_centroid(o_triangles[i], triang.triangles, triang.vertices);
393 double angle_diff = abs(atan2(centroid.y-y_, centroid.x-x_)-yaw_);
394 double corrected_angle_diff = std::min(angle_diff, 2*M_PI-angle_diff);
395
396 // Back edge is behind the car (angle diff > pi/2)
397 if (corrected_angle_diff > M_PI/2) {
398 for (int j = 0; j<3; j++){
399 if (back_edge.size()==2 || triang.triangles[o_triangles[i]].vertices[j] == orig_index){
400 continue;
401 }
402 if (pcl_cloud_[triang.triangles[o_triangles[i]].vertices[j]].color != UNCOLORED){
403 back_edge.push_back(pcl_cloud_[triang.triangles[o_triangles[i]].vertices[j]]);
404 }
405 }
406 if (back_edge.size() != 2) back_edge.clear();
407 if (back_edge.size() == 2 && back_edge[0].color == back_edge[1].color){
408 back_edge.clear();
409 }
410 }
411 }
412
413 return back_edge;
414}

416std::vector<ConeXYZColorScore> PathPlanning::get_final_route(){
417 // Create candidate to final route from cost calculations
418 std::vector<ConeXYZColorScore> final_route = previous_midpoint_routes_.back();
419
420 // Return the final route if there are not enough previous routes
421 if (previous_midpoint_routes_.size() < kRouteBack+1){
422 return final_route;
423 }
424
425 // Create a list of the last routes
426 std::vector<std::vector<ConeXYZColorScore>> last_routes(previous_midpoint_routes_.end()-kRouteBack-1,
427 previous_midpoint_routes_.end()-1);
428
429 // Count the number of points in the last routes that are in the final route
430 int route_count = 0;
431 int total_points = 0;
432 for (int i = 0; i<kRouteBack; i++){
433 int point_count = compare_lists(last_routes[i], final_route, 0.25);
434 route_count += point_count;
435 total_points += last_routes[i].size();
436 }
437
438 // Create threshold to validate last route based on previous route lengths
439 double threshold = total_points*kPrevRouteBias;
440 // If the route is validated, return it and add to previous routes.
441 if (route_count > threshold){
442 invalid_counter_ = 0;
443 return final_route;
444 } else { // Otherwise, return the previous route
445 invalid_counter_++;
446 return previous_midpoint_routes_[previous_midpoint_routes_.size() - 1 - invalid_counter_];
447 }

450common_msgs::msg::Trajectory PathPlanning::create_trajectory_msg(
451 std::vector<ConeXYZColorScore> route, bool smooth){
452 int route_size = route.size();
453 int degree = 2;
454 common_msgs::msg::Trajectory trajectory_msg;
455 double acum = 0.0;
456
457 std::vector<double> xp, yp, xpp, ypp, v_grip, s, k, speed_profile, acc_profile;
458
459 if (route_size <= 3){
460 return trajectory_msg;
461 } else if (!smooth){
462 for (int i = 0; i<route_size; i++){
463 common_msgs::msg::PointXY point;
464 point.x = route[i].x;
465 point.y = route[i].y;
466 trajectory_msg.points.push_back(point);
467 }
468 return trajectory_msg;
469 }
470
471 // Avoid division by zero
472 for (auto p : route){
473 if (std::abs(p.x)<0.001) p.x = 0.001;
474 if (std::abs(p.y)<0.001) p.y = 0.001;
475 }
476
477 // Perform laplacian smoothing previous to spline fitting
478 std::vector<double> new_x, new_y;
479 new_x.push_back(route[0].x);
480 new_y.push_back(route[0].y);
481 for (int i = 1; i<route_size-1;i++){
482 new_x.push_back(((3-kSmoothCoeff)/2*route[i-1].x+kSmoothCoeff*route[i].x+(3-kSmoothCoeff)/2*route[i+1].x)/3);
483 new_y.push_back(((3-kSmoothCoeff)/2*route[i-1].y+kSmoothCoeff*route[i].y+(3-kSmoothCoeff)/2*route[i+1].y)/3);
484 }
485 new_x.push_back(route[route_size-1].x);
486 new_y.push_back(route[route_size-1].y);
487
488 Eigen::MatrixXd control_points(2, route_size);
489 for (int i = 0; i<route_size; i++){
490 control_points(0, i) = new_x[i];
491 control_points(1, i) = new_y[i];
492 }
493
494 Eigen::Spline<double, 2> b_spline = Eigen::SplineFitting<Eigen::Spline<double, 2>>::Interpolate(
495 control_points, degree);
496
497 // Add the points to the trajectory message
498 for (double i = 0; i<route_size*10-1; i++){
499 Eigen::Vector2d point_1 = b_spline(i/(10*route_size));
500 Eigen::Vector2d point_2 = b_spline((i+1)/(10*route_size));
501 double x = point_1(0);
502 double y = point_1(1);
503 double dx = point_2(0)-x;
504 double dy = point_2(1)-y;
505 double dist = hypot(dx, dy);
506 common_msgs::msg::PointXY point;
507 point.x = x;
508 point.y = y;
509 trajectory_msg.points.push_back(point);
510 xp.push_back(dx);
511 yp.push_back(dy);
512 acum += dist;
513 s.push_back(acum);
514 speed_profile.push_back(0.0);
515 }
516
517 xp.push_back(xp.back());
518 yp.push_back(yp.back());
519
520 for (int i = 0; i<xp.size()-1; i++){
521 xpp.push_back(xp[i+1]-xp[i]);
522 ypp.push_back(yp[i+1]-yp[i]);
523 }
524 xpp.push_back(xpp.back());
525 ypp.push_back(ypp.back());
526
527 for (int i = 0; i<xpp.size(); i++){
528 double den = pow(pow(xp[i],2)+pow(yp[i],2), 1.5);
529 double val;
530 if (std::abs(den) < 0.001){
531 val = 0;
532 } else{
533 val = (xp[i]*ypp[i]-yp[i]*xpp[i])/den;
534 }
535 k.push_back(val);
536 }
537
538 for (const auto &val : k){
539 double v = std::min(kMaxVel, sqrt(kMaxYAcc/std::max(abs(val), 0.001)));
540 v_grip.push_back(v);
541 }
542
543 speed_profile[0] = std::max(v_, 0.5);
544
545 for (int i = 1; i<speed_profile.size(); i++){
546 double ds = s[i]-s[i-1];
547 speed_profile[i] = sqrt(pow(speed_profile[i-1], 2) + 2*kMaxXAcc*ds);
548 if (speed_profile[i] > v_grip[i]){
549 speed_profile[i] = v_grip[i];
550 }
551 }
552
553 for (int i = speed_profile.size()-2; i>=0; i--){
554 double ds = s[i+1]-s[i];
555 double v_brake = sqrt(pow(speed_profile[i+1], 2) + 2*kMaxXAcc*ds);
556 if (v_brake < speed_profile[i]){
557 speed_profile[i] = v_brake;
558 }
559 }
560
561 acc_profile.push_back(0.0);
562 for (int i = 0; i<speed_profile.size()-1; i++){
563 double ds = s[i+1]-s[i];
564 if (std::abs(ds) > 0.001){
565 acc_profile.push_back((pow(speed_profile[i+1], 2)-pow(speed_profile[i], 2))/(2*ds));
566 } else {
567 acc_profile.push_back(0.0);
568 }
569 }
570
571 for (int i = 0; i<s.size(); i++){
572 if (std::isnan(speed_profile[i])) speed_profile[i] = 0.0;
573 if (std::isnan(acc_profile[i])) acc_profile[i] = 0.0;
574
575 trajectory_msg.k.push_back(k[i]);
576 trajectory_msg.s.push_back(s[i]);
577 trajectory_msg.speed_profile.push_back(speed_profile[i]);
578 trajectory_msg.acc_profile.push_back(acc_profile[i]);
579 }
580
581 return trajectory_msg;
582}

618common_msgs::msg::TrackLimits PathPlanning::create_track_limits_msg(CDT::TriangleVec triang,
619 std::vector<int> triangles_route){
620 CDT::Triangle last_triangle = triang[triangles_route.back()];
621 CDT::NeighborsArr3 neighbors = last_triangle.neighbors;
622 for (int i = 0; i<3; i++){
623 if (neighbors[i] < triang.size() && !in(neighbors[i], triangles_route)){
624 triangles_route.push_back(neighbors[i]);
625 }
626 }
627 std::vector<int> left_limit, right_limit;
628 for (int i = 0; i<triangles_route.size(); i++){
629 CDT::Triangle triangle = triang[triangles_route[i]];
630 CDT::VerticesArr3 vertices = triangle.vertices;
631 for (int j = 0; j<3; j++){
632 ConeXYZColorScore cone = pcl_cloud_.points[vertices[j]];
633 if (cone.score == -1){
634 continue;
635 }
636 int vertex = vertices[j];
637 if (in(vertex, left_limit) || in(vertex, right_limit)){
638 continue;
639 }
640 switch (cone.color){
641 case BLUE:
642 left_limit.push_back(vertex);
643 break;
644 case YELLOW:
645 right_limit.push_back(vertex);
646 break;
647 case UNCOLORED:
648 if (i < triangles_route.size()-1) {
649 std::vector<double> yaw_vector, cone_vector;
650 CDT::V2d<double> centroid_1 = compute_centroid(triangles_route[i], triang, vertices_);
651 CDT::V2d<double> centroid_2 = compute_centroid(triangles_route[i+1], triang, vertices_);
652 double route_angle = atan2(centroid_2.y-centroid_1.y, centroid_2.x-centroid_1.x);
653 yaw_vector = {cos(route_angle), sin(route_angle)};
654 cone_vector = {cone.x-centroid_1.x, cone.y-centroid_1.y};
655 bool is_positive_angle = yaw_vector[0]*cone_vector[1]-yaw_vector[1]*cone_vector[0] > 0;
656 if (is_positive_angle){
657 left_limit.push_back(vertex);
658 } else {
659 right_limit.push_back(vertex);
660 }
661 } else {
662 std::vector<double> yaw_vector, cone_vector;
663 CDT::V2d<double> centroid_1 = compute_centroid(triangles_route[i], triang, vertices_);
664 CDT::V2d<double> centroid_2 = CDT::V2d<double>::make(1.5, 0);
665 double route_angle = atan2(centroid_2.y-centroid_1.y, centroid_2.x-centroid_1.x);
666 yaw_vector = {cos(route_angle), sin(route_angle)};
667 cone_vector = {cone.x-centroid_1.x, cone.y-centroid_1.y};
668 bool is_positive_angle = yaw_vector[0]*cone_vector[1]-yaw_vector[1]*cone_vector[0] > 0;
669 if (is_positive_angle){
670 left_limit.push_back(vertex);
671 } else {
672 right_limit.push_back(vertex);
673 }
674 }
675 break;
676 }
677 }
678 }
679 common_msgs::msg::TrackLimits track_limits_msg;
680 for (int i = 0; i<left_limit_.size(); i++){
681 common_msgs::msg::PointXY point;
682 point.x = left_limit_[i].x;
683 point.y = left_limit_[i].y;
684 track_limits_msg.left_limit.push_back(point);
685 }
686 for (int i = 0; i<left_limit.size(); i++){
687 common_msgs::msg::PointXY point;
688 point.x = pcl_cloud_.points[left_limit[i]].x;
689 point.y = pcl_cloud_.points[left_limit[i]].y;
690 track_limits_msg.left_limit.push_back(point);
691 }
692 for (int i = 0; i<right_limit_.size(); i++){
693 common_msgs::msg::PointXY point;
694 point.x = right_limit_[i].x;
695 point.y = right_limit_[i].y;
696 track_limits_msg.right_limit.push_back(point);
697 }
698 for (int i = 0; i<right_limit.size(); i++){
699 common_msgs::msg::PointXY point;
700 point.x = pcl_cloud_.points[right_limit[i]].x;
701 point.y = pcl_cloud_.points[right_limit[i]].y;
702 track_limits_msg.right_limit.push_back(point);
703 }
704 track_limits_msg.trajectory = this->create_trajectory_msg(closing_route_, false);
705
706 return track_limits_msg;
707
708}

711 common_msgs::msg::Triangulation triangulation_msg;
712 CDT::Triangulation<double>::V2dVec points = triangulation.vertices;
713 CDT::TriangleVec triangles = triangulation.triangles;
714 for (int i = 0; i < triangles.size(); i++){
715 common_msgs::msg::Simplex simplex;
716 common_msgs::msg::PointXY a, b, c;
717 CDT::Triangle triangle = triangles[i];
718 CDT::VerticesArr3 vertices_index = triangle.vertices;
719 CDT::VertInd a_ind = vertices_index[0];
720 CDT::VertInd b_ind = vertices_index[1];
721 CDT::VertInd c_ind = vertices_index[2];
722 a.x = points[a_ind].x;
723 a.y = points[a_ind].y;
724 b.x = points[b_ind].x;
725 b.y = points[b_ind].y;
726 c.x = points[c_ind].x;
727 c.y = points[c_ind].y;
728 simplex.points = {a, b, c};
729 triangulation_msg.simplices.push_back(simplex);
730 }
731 return triangulation_msg;
