56void InspectionControl::on_timer()
57{
58 rclcpp::Time t0 = this->now();
59
60 if(as_status_ == 3 && this->get_clock()->now().seconds() - start_time_.seconds() < kDuration){
61 auto cmd_msg = common_msgs::msg::Cmd();
62
63 // 230 is the maximum toque value
64 // Clamp the value between 0% and 20% of the maximum torque
65 cmd_msg.acc = std::clamp(pid_.compute_control(vx_, 1.0, 0.01), 0.0, 3.0);
66
67 double t = this->get_clock()->now().seconds() - start_time_.seconds();
68 cmd_msg.delta = (kAmplitude*M_PI/180)*sin(2.0*M_PI*kFrequency*t);
69
70 cmd_pub_->publish(cmd_msg);
71
72 if (kDebug) {
73 RCLCPP_INFO(this->get_logger(), "Duration: %f", t);
74 RCLCPP_INFO(this->get_logger(), "Acceleration: %f", cmd_msg.acc);
75 RCLCPP_INFO(this->get_logger(), "Steering angle: %f", cmd_msg.delta);
76 }
77
78 } else if (as_status_ == 3 && this->get_clock()->now().seconds() - start_time_.seconds() >= kDuration){
79 if (vx_ > 0.5){ // When duration is over, stop accelerating until finish
80 auto cmd_msg = common_msgs::msg::Cmd();
81 cmd_msg.acc = 0.0;
82 cmd_msg.delta = 0.0;
83 cmd_pub_->publish(cmd_msg);
84 if (kDebug) RCLCPP_INFO(this->get_logger(), "Braking");
85 } else {
86 auto finish_msg = std_msgs::msg::Float32();
87 finish_msg.data = 5;
88 finish_pub_->publish(finish_msg);
89 if (kDebug) RCLCPP_INFO(this->get_logger(), "Finished");
90 }
91 }
92
93 if (kDebug) RCLCPP_INFO(this->get_logger(), "Total time: %f", this->now().seconds() - t0.seconds());
94}
