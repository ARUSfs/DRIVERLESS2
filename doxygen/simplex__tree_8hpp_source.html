96
97 SimplexNode* node = new SimplexNode(index);
98 if (midpoint_routes_.size() > 3 and route_cost > 600*len_coeff_){
99 // Best route cost will never be higher than 500 m times the length coeff plus the min cost
100 return node;
101 }
102 CDT::Triangle triangle = triangle_list[index]; // Get the triangle from the index
103 CDT::NeighborsArr3 neighbors = triangle.neighbors; // Get the neighbors of the triangle
104 std::vector<ConeXYZColorScore> neighbor_edge = {};
105 std::vector<int> valid_neighbors = {};
106 for (int i = 0; i<3; i++){
107 if (neighbors[i]>triangle_list.size()) continue;
108
109 CDT::Triangle n_triangle = triangle_list[neighbors[i]];
110 std::vector<ConeXYZColorScore> next_edge = {};
111
112 bool finish_route = false;
113
114 for (auto v: n_triangle.vertices){
115 if (!in(v, triangle.vertices)){
116 if (in(v, passed_vertices)){ // Check the neighbor doesn't contain any passed vertex
117 if (passed_vertices.size() > cones_cloud_.size()*0.5 && (v == passed_vertices[0] || v == passed_vertices[1])){ // If the first vertex is reached again, end the route
118 finish_route = true;
119 } else {
120 next_edge.clear();
121 break;
122 }
123 }
124 } else {
125 next_edge.push_back(cones_cloud_.points[v]);
126 }
127 }
128
129 if (finish_route && next_edge.size() == 2){
130 auto v = visited;
131 v.push_back(neighbors[i]);
132 index_routes_.push_back(v);
133 auto m = mid_route;
134 m.push_back(ConeXYZColorScore((next_edge[0].x+next_edge[1].x)/2,
135 (next_edge[0].y+next_edge[1].y)/2, 0, UNCOLORED, 1));
136 midpoint_routes_.push_back(m);
137 ending_routes_.push_back(m);
138 if (m.size() > 0.95*cones_cloud_.size()){
139 route_cost -= 100;
140 }

142 min_cost_ = route_cost;
143 best_route_ = mid_route;
144 best_index_route_ = visited;
145 }

147 }
148

150 (next_edge[0].color == UNCOLORED ||
151 next_edge[1].color == UNCOLORED ||
152 next_edge[0].color != next_edge[1].color) && // Check the edge is not colored or is colored with different colors

154 valid_neighbors.push_back(neighbors[i]);
155 neighbor_edge.push_back(ConeXYZColorScore((next_edge[0].x+next_edge[1].x)/2,
156 (next_edge[0].y+next_edge[1].y)/2, 0, UNCOLORED, 1));
157 }
158
159 }
160
161

163 if (valid_neighbors.size() == 1){
164 // In case only one neighbor is valid, create the left child and return the node
165 visited.push_back(valid_neighbors[0]);
166 ConeXYZColorScore prev_midpoint = mid_route.back();
167 mid_route.push_back(neighbor_edge[0]);
168
169 // Cost calculations

171 double angle_diff = abs(angle-prev_angle);
172 double corr_angle_diff = std::min(angle_diff, 2*M_PI-angle_diff);
173 double prev_angle = angle; // Save the previous angle for the next iteration
174 double len = distance(neighbor_edge[0], prev_midpoint);
175 if (corr_angle_diff > M_PI/6){
176 route_cost += angle_coeff_*(3*pow(corr_angle_diff-M_PI/6, 2)+1);
177 }
178 route_cost -= len_coeff_*len;
179
180
181 CDT::Triangle next_triangle = triangle_list[valid_neighbors[0]];
182 for (auto v: triangle.vertices){
183 if (!in(v, next_triangle.vertices)){
184 passed_vertices.push_back(v);
185 }
186 }

188 node->left = SimplexTree::create_tree_aux(triangle_list, valid_neighbors[0], visited,
189 passed_vertices, mid_route, route_cost, prev_angle);
190 return node;
191 }
192 else if (valid_neighbors.size() == 2){
193 // Store the previous midpoint to calculate the angle between the following midpoints
194 ConeXYZColorScore prev_midpoint = mid_route.back();
195
196 // In case two neighbors are valid, create both children and return the node
197 std::vector<int> visited_left = visited;
