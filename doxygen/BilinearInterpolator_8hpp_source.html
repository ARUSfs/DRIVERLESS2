87template <class Real>
88void BilinearInterpolator<Real>::setupInterpolator() {
89 // Interpolation will be done by multiplying the coordinates by
90 // coefficients.
91
92 Q = Matrix22Array(X->size() - 1, Y->size() - 1);
93
94 // We are going to pre-compute the interpolation coefficients so
95 // that we can interpolate quickly
96 for (int i = 0; i < X->size() - 1; i++) {
97 for (int j = 0; j < Y->size() - 1; j++) {
98 Real tmp = (((*X)(i + 1) - (*X)(i)) * ((*Y)(j + 1) - (*Y)(j)));
99 Q(i, j) = Z->block(i, j, 2, 2) / tmp;
100 }
101 }
102}

104template <class Real>
105Real BilinearInterpolator<Real>::operator()(Real x, Real y) const {
106 BASE::checkData();
107
108 // no extrapolation...
109 if (x < (*X)(0) || x > (*X)(X->size() - 1) || y < (*Y)(0) ||
110 y > (*Y)(Y->size() - 1)) {
111 return 0;
112 }
113
114 int i = this->get_x_index_to_left_of(x);
115 int j = this->get_y_index_below(y);
116
117 if (i < 0) i = 0;
118 if (j < 0) j = 0;
119
120 // now, create the coordinate vectors (see Wikipedia
121 // https://en.wikipedia.org/wiki/Bilinear_interpolation)
122 RowVector2 vx;
123 ColVector2 vy;
124 vx << ((*X)(i + 1) - x), (x - (*X)(i));
125 vy << ((*Y)(j + 1) - y), (y - (*Y)(j));
126
127 // interpolation is just x*Q*y
128
129 return vx * Q(i, j) * vy;
130}
