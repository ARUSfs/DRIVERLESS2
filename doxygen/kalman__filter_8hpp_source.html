238 // PREDICTION STAGE
239 // Transition matrix
240 std::chrono::steady_clock::time_point now = std::chrono::steady_clock::now();
241 double dt = std::chrono::duration_cast<std::chrono::nanoseconds>(now - t_).count() * pow(10, -9);
242
243 MatrixXd A(n_, n_);
244 A = MatrixXd::Identity(n_, n_) + dt * M_;
245
246 // State prediction
247 VectorXd x_pred = A * x_ + dt * B_ * u_;
248
249 // Covariance prediction
250 MatrixXd P_pred = A * P_ * A.transpose() + Q_;
251
252 // INNOVATION
253 VectorXd v = z - H_ * x_pred;
254
255 // KALMAN GAIN
256 MatrixXd K = P_pred * H_.transpose() * (H_ * P_pred * H_.transpose() + R_).inverse();
257
258 // UPDATE STAGE
259 // State estimation
260 VectorXd x_est = x_pred + K * v;
261
262 // Covariance estimation
263 MatrixXd P_est = (MatrixXd::Identity(n_, n_) - K * H_) * P_pred;
264
265 // Save estimated data for the next iteration
266 t_ = std::chrono::steady_clock::now();
267 x_ = x_est;
268 P_ = P_est;
269 u_ = u;
