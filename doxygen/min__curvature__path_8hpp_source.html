48 pair<MatrixXd,qpmad::Solver::ReturnStatus> get_min_curvature_path(VectorXd x, VectorXd y, VectorXd twr, VectorXd twl, int n_seg){
49 //First, we process track data
50 MatrixXd track_data = process_track_data(x, y, twr, twl, n_seg);
51
52 //Then, we form the matrices which will define the quadratic optimization problem
53 VectorXd xin = track_data.col(2);
54 VectorXd yin = track_data.col(3);
55 VectorXd xout = track_data.col(4);
56 VectorXd yout = track_data.col(5);
57
58 VectorXd delx = xout - xin;
59 VectorXd dely = yout - yin;
60
61 MatrixXd H = matrixH(delx, dely);
62 VectorXd B = vectorB(xin, yin, delx, dely);
63
64 //Solve the quadratic problem
65 auto solver_out = qpmad_solver(H,B,xin, yin, delx, dely);
66 VectorXd resMCP = solver_out.first;
67 qpmad::Solver::ReturnStatus status = solver_out.second;
68
69 int m = resMCP.size();
70 MatrixXd res(m,2);
71
72 if(status == qpmad::Solver::OK){
73 //Coordinates for the resultant curve
74 VectorXd xresMCP = VectorXd::Zero(m);
75 VectorXd yresMCP = VectorXd::Zero(m);
76
77 for(int i = 0; i < m; i++){
78 xresMCP(i) = xin(i) + resMCP(i)*delx(i);
79 yresMCP(i) = yin(i) + resMCP(i)*dely(i);
80 }
81
82 res << xresMCP, yresMCP;
83 }
84
85 return {res,status};
86 }

88 MatrixXd process_track_data(VectorXd x, VectorXd y, VectorXd twr, VectorXd twl, int n_seg){
89 //Interpolate path to get finer curve with equidistant points
90 int n = x.size();
91
92 MatrixXd path_x_y(n,2); path_x_y << x, y;
93
94 MatrixXd difs2 = diff_col(path_x_y).array().square().matrix();
95 VectorXd step_lengths = (difs2.col(0)+difs2.col(1)).cwiseSqrt();
96
97 VectorXd temp_step = step_lengths;
98 step_lengths.conservativeResize(step_lengths.size()+1);
99 step_lengths << 0, temp_step; // add the starting point
100
101 VectorXd cumulative_len = cumsum(step_lengths);
102 VectorXd final_step_locks = VectorXd::LinSpaced(n_seg, 0, cumulative_len(cumulative_len.size()-1));
103
104 int m = final_step_locks.size();
105
106 MatrixXd final_path_x_y(m, 2);
107 _1D::LinearInterpolator<double> interp;
108 interp.setData(cumulative_len, x);
109 for(int i = 0; i < m; i++){
110 final_path_x_y(i, 0) = interp(final_step_locks(i));
111 }
112
113 interp.setData(cumulative_len, y);
114 for(int i = 0; i < m; i++){
115 final_path_x_y(i, 1) = interp(final_step_locks(i));
116 }
117
118 VectorXd xt = final_path_x_y.col(0);
119 VectorXd yt = final_path_x_y.col(1);
120
121 // Interpolate track widths too
122 _1D::CubicSplineInterpolator<double> interp_c;
123 interp_c.setData(cumulative_len, twr);
124 VectorXd twrt(m);
125 for(int i = 0; i < m; i++){
126 twrt(i) = interp_c(final_step_locks(i));
127 }
128
129 interp_c.setData(cumulative_len, twl);
130 VectorXd twlt(m);
131 for(int i = 0; i < m; i++){
132 twlt(i) = interp_c(final_step_locks(i));
133 }
134
135 // Normal direction for each vertex
136 VectorXd dx = gradient(xt);
137 auto dx_a = dx.array();
138 VectorXd dy = gradient(yt);
139 auto dy_a = dy.array();
140 VectorXd dL = (dx_a*dx_a + dy_a*dy_a).sqrt().matrix();
141
142 // Get inner and outer curves
143 MatrixXd offset(m, 2);
144 offset << -twrt, twlt;
145 VectorXd xin = VectorXd::Zero(m);
146 VectorXd yin = VectorXd::Zero(m);
147 VectorXd xout = VectorXd::Zero(m);
148 VectorXd yout = VectorXd::Zero(m);
149
150 VectorXd aux_x = (dy.array()/dL.array()).matrix();
151 VectorXd aux_y = (dx.array()/dL.array()).matrix();
152
153 for(int i = 0; i < m; i++){
154 VectorXd xinv, yinv, xoutv, youtv;
155
156 xinv = -offset(i,0)*aux_x + xt; //get inner offset curve
157 yinv = offset(i,0)*aux_y + yt;
158
159 xoutv = -offset(i,1)*aux_x + xt; //get outer offset curve
160 youtv = offset(i,1)*aux_y + yt;
161
162 xin(i) = xinv(i);
163 yin(i) = yinv(i);
164 xout(i) = xoutv(i);
165 yout(i) = youtv(i);
166 }
167
168 MatrixXd track_data(m,6);
169 track_data << xt, yt, xin, yin, xout, yout;
170
171 return track_data;
172 }

174 MatrixXd matrixH(VectorXd delx, VectorXd dely){
175 int n = delx.size();
176 MatrixXd H = MatrixXd::Zero(n,n);
177
178 for(int i = 1; i < n-1; i++ ){
179 //First row
180 H(i-1,i-1) = H(i-1,i-1) + pow(delx(i-1),2) + pow(dely(i-1),2);
181 H(i-1,i) = H(i-1,i) - 2*delx(i-1)*delx(i) - 2*dely(i-1)*dely(i);
182 H(i-1,i+1) = H(i-1,i+1) + delx(i-1)*delx(i+1) + dely(i-1)*dely(i+1);
183
184 //Second row
185 H(i,i-1) = H(i,i-1) - 2*delx(i-1)*delx(i) - 2*dely(i-1)*dely(i);
186 H(i,i) = H(i,i ) + 4*pow(delx(i),2) + 4*pow(dely(i),2);
187 H(i,i+1) = H(i,i+1) - 2*delx(i)*delx(i+1) - 2*dely(i)*dely(i+1);
188
189 //Third row
190 H(i+1,i-1) = H(i+1,i-1) + delx(i-1)*delx(i+1) + dely(i-1)*dely(i+1);
191 H(i+1,i) = H(i+1,i) - 2*delx(i)*delx(i+1) - 2*dely(i)*dely(i+1);
192 H(i+1,i+1) = H(i+1,i+1) + pow(delx(i+1),2) + pow(dely(i+1),2);
193 }
194
195 return H;
196 }

199 int n = delx.size();
200 VectorXd B = VectorXd::Zero(n);
201
202 for(int i = 1; i < n-1; i++){
203 B(i-1) = B(i-1) + 2*(xin(i+1)+xin(i-1)-2*xin(i))*delx(i-1) + 2*(yin(i+1)+yin(i-1)-2*yin(i))*dely(i-1);
204 B(i) = B(i) - 4*(xin(i+1)+xin(i-1)-2*xin(i))*delx(i) - 4*(yin(i+1)+yin(i-1)-2*yin(i))*dely(i);
205 B(i+1) = B(i+1) + 2*(xin(i+1)+xin(i-1)-2*xin(i))*delx(i+1) + 2*(yin(i+1)+yin(i-1)-2*yin(i))*dely(i+1);
206 }

211 pair<VectorXd,qpmad::Solver::ReturnStatus> qpmad_solver(MatrixXd H, VectorXd B, VectorXd xin, VectorXd yin, VectorXd delx, VectorXd dely){
212 // Define lower and upper bounds
213 int n = H.rows();
214 VectorXd lb = VectorXd::Zero(n);
215 VectorXd ub = VectorXd::Ones(n);
216
217 // Define equality constraints (same first and last point with smooth transition)
218 // Solver doesn't handle equality constraints, but we can implement it as two inequalities:
219 // beq <= Aeq*res <= beq
220 MatrixXd Aeq = MatrixXd::Zero(3,n);
221 Aeq(0,0) = delx(0);
222 Aeq(0,1) = -delx(1);
223 Aeq(0,n-2) = -delx(n-2);
224 Aeq(0,n-1) = delx(n-1);
225
226 Aeq(1,0) = dely(0);
227 Aeq(1,1) = -dely(1);
228 Aeq(1,n-2) = -dely(n-2);
229 Aeq(1,n-1) = dely(n-1);
230
231 Aeq(2,0) = -1;
232 Aeq(2,n-1) = 1;
233
234 VectorXd beq(3);
235 beq << -xin(0)+xin(1)+xin(n-2)-xin(n-1), -yin(0)+yin(1)+yin(n-2)-yin(n-1), 0;
236
237 VectorXd Alb(3), Aub(3);
238 Alb << beq;
239 Aub << beq;
240
241 // Declare solution and solver
242 VectorXd res(n);
243 qpmad::Solver solver;
244
245 // Solve optimization problem
246 H << 2*H;
247 qpmad::Solver::ReturnStatus status = solver.solve(res, H, B, lb, ub, Aeq, Alb, Aub);
248
249 return {res,status};
250 }
